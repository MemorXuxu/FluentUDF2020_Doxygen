17{ if (r < 0. || r > 1.0)\
18 f = u;\
19 else if (r >= b)\
20 f = c;\
21 else\
22 {real g = r/b;\
23 f = g*c+(1.-g)*v;}\
24}

26#define WBCD_LIMITER(f,r,c,u,v,alfa,cslope)\
27 { real beta1 = 0.5-0.4/MAX(alfa,1.0e-30);\
28 real beta2 = 0.5+0.4/MAX(alfa,1.0e-30);\
29 real g = 1.0;\
30 if(r <0.5)\
31 g = 0.5*(1.0+tanh(cslope*(r-beta1)));\
32 else if(r >= 0.5 && r<= 1.0)\
33 g = 1;\
34 else\
35 {g = 0.5*(1.0-tanh(cslope*(r-beta2)));}\
36 f = g*c+(1.-g)*v;\
37}

40 f = u;\

42 f = c;\
43 else\
44 {real g = r*r/b/b;\
45 f = g*c+(1.-g)*v;}\
