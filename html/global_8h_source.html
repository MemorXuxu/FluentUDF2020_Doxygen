277/* Wrappers for disabling/restoring interrupts.
278 * Insures that an inner scope does not enable interrupts when
279 * and outer scope has disabled them.
280 * Usage/format (the braces are required):
281 *
282 * {
283 * DISABLE_INTERRUPTS();
284 *
285 * statement;
286 * ...
287 * if (error)
288 * {
289 * RESTORE_INTERRUPTS();
290 * return;
291 * }
292 * statement;
293 *
294 * RESTORE_INTERRUPTS();
295 * }
296 */
297
298#define DISABLE_INTERRUPTS() \
299 volatile cxboolean _interrupts_were_on = interrupts_enabled(); \
300 if (_interrupts_were_on) disable_interrupts()
301
302#define RESTORE_INTERRUPTS() \
303 if (_interrupts_were_on) enable_interrupts()
304
305#define fl_interrupted CX_Check_Interrupt()
306
307/* allow alternate ways of disabling/enabling interrupts. Eg.:
308 *
309 * {
310 * Init_Interrupts();
311 * if (X)
312 * Disable_Interrupts();
313 *
314 * ...
315 * ...
316 *
317 * if (X)
318 * Restore_Interrupts();
319 * }
320 */
321
322#define Init_Interrupts() \
323 cxboolean _interrupts_were_on = FALSE
324
325#define Disable_Interrupts() \
326 _interrupts_were_on = interrupts_enabled(); \
327 if (_interrupts_were_on) disable_interrupts()
328
329#define Restore_Interrupts() \
330 if (_interrupts_were_on) enable_interrupts()
