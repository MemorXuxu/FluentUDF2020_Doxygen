348/*
349 * C99 specifies that complex numbers have the same representation as
350 * an array of two elements, where the first element is the real part
351 * and the second element is the imaginary part.
352 */
353#define __NPY_CPACK_IMP(x, y, type, ctype) \
354 union { \
355 ctype z; \
356 type a[2]; \
357 } z1;; \
358 \
359 z1.a[0] = (x); \
360 z1.a[1] = (y); \
361 \
362 return z1.z;
363
364static NPY_INLINE npy_cdouble npy_cpack(double x, double y)
365{
366 __NPY_CPACK_IMP(x, y, double, npy_cdouble);
367}

380/*
381 * Same remark as above, but in the other direction: extract first/second
382 * member of complex number, assuming a C99-compatible representation
383 *
384 * Those are defineds as static inline, and such as a reasonable compiler would
385 * most likely compile this to one or two instructions (on CISC at least)
386 */
387#define __NPY_CEXTRACT_IMP(z, index, type, ctype) \
388 union { \
389 ctype z; \
390 type a[2]; \
391 } __z_repr; \
392 __z_repr.z = z; \
393 \
394 return __z_repr.a[index];
395
396static NPY_INLINE double npy_creal(npy_cdouble z)
397{
398 __NPY_CEXTRACT_IMP(z, 0, double, npy_cdouble);
399}
