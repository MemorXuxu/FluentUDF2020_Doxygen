186#define Reallocate_Stack() \
187{ \
188 size_t origOffset = stack_now - stack_start; \
189 stack_size += default_stack_size; \
190 stack_start = (stack *)realloc(stack_start, \
191 sizeof(stack)*stack_size); \
192 stack_now = stack_start + origOffset; \
193 stack_end = stack_start + stack_size; \
194}
195
196#define Push_Stack(_x) \
197{ \
198 if(stack_now==stack_end) { \
199 Reallocate_Stack(); \
200 } \
201 *stack_now++ = &(_x); \
202 (_x)=NIL; \
203}
204
205#define Push_Stack0(_x) \
206{ \
207 if(stack_now==stack_end) { \
208 Reallocate_Stack(); \
209 } \
210 *stack_now++ = &(_x); \
211}
