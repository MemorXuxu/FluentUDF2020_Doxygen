13#define SET_K_EFF(c0,t0,c1,t1,k0,k1)\
14{\
15 if (GVAR_TURB(rp, ke_rng) || GVAR_TURB(rp, les_rng))\
16 {\
17 if( M_thickened_flame )\
18 {\
19 k0 = C_RNG_ALPHA(c0,t0)*(C_MU_L(c0,t0)+C_TFM_OMEGA(c0,t0)*C_MU_T(c0,t0))*C_CP(c0,t0);\
20 k1 = C_RNG_ALPHA(c1,t1)*(C_MU_L(c1,t1)+C_TFM_OMEGA(c1,t1)*C_MU_T(c1,t1))*C_CP(c1,t1);\
21 }\
22 else\
23 {\
24 k0 = C_RNG_ALPHA(c0,t0)*C_MU_EFF(c0,t0)*C_CP(c0,t0);\
25 k1 = C_RNG_ALPHA(c1,t1)*C_MU_EFF(c1,t1)*C_CP(c1,t1);\
26 }\
27 }\
28 else if (GVAR_TURB(rp, kklw))\
29 {\
30 real rho0 = C_R(c0,t0);\
31 real rho1 = C_R(c1,t1);\
32 real fac = (real)(2./3.);\
33 real lambda0_T = sqrt(C_KT(c0,t0))/C_O(c0,t0);\
34 real lambda1_T = sqrt(C_KT(c1,t1))/C_O(c1,t1);\
35 real lambda0_eff = MIN(GVAR_TURB(coeff, trans_Clam)*C_WALL_DIST(c0,t0), lambda0_T);\
36 real lambda1_eff = MIN(GVAR_TURB(coeff, trans_Clam)*C_WALL_DIST(c1,t1), lambda1_T);\
37 real lambda0_fac = pow(lambda0_eff/lambda0_T, fac);\
38 real lambda1_fac = pow(lambda1_eff/lambda1_T, fac);\
39 k0 = (lambda0_fac*(C_KT(c0,t0)/(C_KL(c0,t0)+C_KT(c0,t0)))*C_MU_TS(c0,t0)/GVAR_TURB(coeff, ke_prt) + \
40 GVAR_TURB(coeff, trans_Calpha_teta)*rho0*(1.-lambda0_fac)*sqrt(C_KT(c0,t0))*lambda0_eff )*C_CP(c0,t0);\
41 k1 = (lambda1_fac*(C_KT(c1,t1)/(C_KL(c1,t1)+C_KT(c1,t1)))*C_MU_TS(c1,t1)/GVAR_TURB(coeff, ke_prt) + \
42 GVAR_TURB(coeff, trans_Calpha_teta)*rho1*(1.-lambda1_fac)*sqrt(C_KT(c1,t1))*lambda1_eff )*C_CP(c1,t1);\
43 if( M_thickened_flame )\
44 {\
45 k0 *= C_TFM_OMEGA(c0,t0);\
46 k1 *= C_TFM_OMEGA(c1,t1);\
47 }\
48 k0 += C_K_L(c0,t0);\
49 k1 += C_K_L(c1,t1);\
50 }\
51 else if (GVAR_TURB(rp, les_dynamic_energy))\
52 {\
53 real Pr0 = C_STORAGE_R(c0,t0,SV_LES_PRANDTL);\
54 real Pr1 = C_STORAGE_R(c1,t1,SV_LES_PRANDTL);\
55 k0 = C_K_EFF(c0,t0,Pr0);\
56 k1 = C_K_EFF(c1,t1,Pr1);\
57 }\
58 else\
59 {\
60 if (user_prandtl)\
61 {\
62 real Pr0 = Compute_User_Prandtl(UDF_PRT_T,c0,t0);\
63 real Pr1 = Compute_User_Prandtl(UDF_PRT_T,c1,t1);\
64 k0 = C_K_EFF(c0,t0,Pr0);\
65 k1 = C_K_EFF(c1,t1,Pr1);\
66 }\
67 else\
68 {\
69 k0 = C_K_EFF(c0,t0,GVAR_TURB(coeff, ke_prt));\
70 k1 = C_K_EFF(c1,t1,GVAR_TURB(coeff, ke_prt));\
71 }\
72 }\
73}

75#define SET_K_EFF_INFLOW(c0,t0,k_eff)\
76{\
77 if (GVAR_TURB(rp, ke_rng) || GVAR_TURB(rp, les_rng))\
78 {\
79 if( M_thickened_flame )\
80 k_eff = C_RNG_ALPHA(c0,t0)*(C_MU_L(c0,t0)+C_TFM_OMEGA(c0,t0)*C_MU_T(c0,t0))*C_CP(c0,t0);\
81 else\
82 k_eff = C_RNG_ALPHA(c0,t0)*C_MU_EFF(c0,t0)*C_CP(c0,t0);\
83 }\
84 else if (GVAR_TURB(rp, kklw))\
85 {\
86 real rho0 = C_R(c0,t0);\
87 real fac = (real)(2./3.);\
88 real lambda0_T = sqrt(C_KT(c0,t0))/C_O(c0,t0);\
89 real lambda0_eff = MIN(GVAR_TURB(coeff, trans_Clam)*C_WALL_DIST(c0,t0), lambda0_T);\
90 real lambda0_fac = pow(lambda0_eff/lambda0_T, fac);\
91 k_eff = (lambda0_fac*(C_KT(c0,t0)/(C_KL(c0,t0)+C_KT(c0,t0)))*C_MU_TS(c0,t0)/GVAR_TURB(coeff, ke_prt) + \
92 GVAR_TURB(coeff, trans_Calpha_teta)*rho0*(1.-lambda0_fac)*sqrt(C_KT(c0,t0))*lambda0_eff) * C_CP(c0,t0) ;\
93 if( M_thickened_flame ) k_eff *= C_TFM_OMEGA(c0,t0);\
94 k_eff += C_K_L(c0,t0);\
95 }\
96 else\
97 {\
98 if (user_prandtl)\
99 {\
100 real Pr0 = Compute_User_Prandtl(UDF_PRT_T,c0,t0);\
101 k_eff = C_K_EFF(c0,t0,Pr0);\
102 }\
103 else\
104 k_eff = C_K_EFF(c0,t0,GVAR_TURB(coeff, ke_prt));\
105 }\
106}
