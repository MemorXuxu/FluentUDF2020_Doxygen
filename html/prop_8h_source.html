370#define DIFFUSION_COEF_AVE_LONG(g,g0,g1,d0,d1,r) \
371 {\
372 real _d;\
373 g = (g0)*(g1);\
374 _d = (d1)*(g0)+(d0)*(g1)+(r)*(g);\
