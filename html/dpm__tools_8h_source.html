106/* NOTE: The following uses MAG(...), which always uses all 3 (three)
107 * components, even for RP_2D (regardless of rp_axi_swirl etc.)!
108 * ---> In 2d, make sure the third component is initialised (e.g. 0.)!
109 */
110#define UNIT_VECT(A,R) {if (VZEROP(A)) R[0] = R[1] = R[2] = 0.0; \
111 else { \
112 real _m; \
113 _m = MAG(A); \
114 R[0] = A[0]/_m; \
115 R[1] = A[1]/_m; \
116 R[2] = A[2]/_m; } }
117

119/* have a smaller limit to check for zero vectors in particular for area vectors. */
120/* NOTE: The following uses MAG(...), which always uses all 3 (three)
121 * components, even for RP_2D (regardless of rp_axi_swirl etc.)!
122 * ---> In 2d, make sure the third component is initialised (e.g. 0.)!
123 */
124#define DPM_UNIT_VECT(A,R) {\
125 if ((ABS(A[0]) > 1e3*REAL_MIN) || \
126 (ABS(A[1]) > 1e3*REAL_MIN) || \
127 (ABS(A[2]) > 1e3*REAL_MIN)) \
128 { \
129 double _m; \
130 _m = 1./MAG(A); \
131 R[0] = A[0]*_m; \
132 R[1] = A[1]*_m; \
133 R[2] = A[2]*_m; \
134 } \
135 else \
136 R[0] = R[1] = R[2] = 0.; \
137 }
138

140/* NOTE: The following uses MAG(...), which always uses all 3 (three)
141 * components, even for RP_2D (regardless of rp_axi_swirl etc.)!
142 * ---> In 2d, make sure the third component is initialised (e.g. 0.)!
143 */
144#define DPM_UNIT_VECT_MAG(A,R,M) \
145{ \
146 if ((ABS(A[0]) > 1e3*REAL_MIN) || \
147 (ABS(A[1]) > 1e3*REAL_MIN) || \
148 (ABS(A[2]) > 1e3*REAL_MIN)) \
149 { \
150 double _m; \
151 M = MAG(A); \
152 _m = 1./M; \
153 R[0] = A[0]*_m; \
154 R[1] = A[1]*_m; \
155 R[2] = A[2]*_m; \
156 } \
157 else \
158 R[0] = R[1] = R[2] = M = 0.; \
159}

296# define FCalc_F_Area_M1(A,f,t)\
297 switch (F_NNODES(f,t))\
298 {\
299 case TRI_FACE: /* 3D triangular face */\
300 {\
301 Node *_n0 = F_NODE(f,t,0);\
302 Node *_n1 = F_NODE(f,t,1);\
303 Node *_n2 = F_NODE(f,t,2);\
304 A[0] = TRI_AREA_X_M1(_n0,_n1,_n2);\
305 A[1] = TRI_AREA_Y_M1(_n0,_n1,_n2);\
306 A[2] = TRI_AREA_Z_M1(_n0,_n1,_n2);\
307 }\
308 break;\
309\
310 case QUAD_FACE: /* 3D quad face */\
311 {\
312 Node *_n0 = F_NODE(f,t,0);\
313 Node *_n1 = F_NODE(f,t,1);\
314 Node *_n2 = F_NODE(f,t,2);\
315 Node *_n3 = F_NODE(f,t,3);\
316 A[0] = QUAD_AREA_X_M1(_n0,_n1,_n2,_n3);\
317 A[1] = QUAD_AREA_Y_M1(_n0,_n1,_n2,_n3);\
318 A[2] = QUAD_AREA_Z_M1(_n0,_n1,_n2,_n3);\
319 }\
320 break;\
321\
322 case MIXED_FACE: /* Accumulated face */\
323 case POLY_FACE:\
324 default: /* 3D polygonal face */\
325 {\
326 register int _i;\
327 \
328 NV_S (A, =, 0.);\
329\
330 for (_i = 0; _i < F_NNODES(f,t); ++_i)\
331 {\
332 Node *_n0 = F_NODE (f, t, _i);\
333 Node *_n1 = F_NODE (f, t, (_i+1)%F_NNODES(f,t));\
334 ND_MAP(A[0],A[1],A[2], +=,\
335 (NODE_Y_M1(_n1) + NODE_Y_M1(_n0))*(NODE_Z_M1(_n1) - NODE_Z_M1(_n0)),\
336 (NODE_Z_M1(_n1) + NODE_Z_M1(_n0))*(NODE_X_M1(_n1) - NODE_X_M1(_n0)),\
337 (NODE_X_M1(_n1) + NODE_X_M1(_n0))*(NODE_Y_M1(_n1) - NODE_Y_M1(_n0)));\
338 }\
339 NV_S (A, /=, 2.);\
340 }\
341 break;\
342 }
