40/* Note we need the 2 concats below because arguments to ##

42 * before doing the actual concatenation. */

44#define ASSERT_CONCAT(a, b) ASSERT_CONCAT_(a, b)
45/* These can't be used after statements in c89. */

47# define STATIC_ASSERT(e,m) \
48 /* ; */ enum { ASSERT_CONCAT(static_assert_, __COUNTER__) = 1/(int)(!!(e)) }

51 * that the headers are not included twice (by wrapping in #ifndef...#endif)
