96/*
97 * ----------------------------------------------------------------
98 * Type : IDAResFn
99 * ----------------------------------------------------------------
100 * The F function which defines the DAE system F(t,y,y')=0
101 * must have type IDAResFn.
102 * Symbols are as follows:
103 * t <-> t y <-> yy
104 * y' <-> yp F <-> rr
105 * A IDAResFn takes as input the independent variable value t,
106 * the dependent variable vector yy, and the derivative (with
107 * respect to t) of the yy vector, yp. It stores the result of
108 * F(t,y,y') in the vector rr. The yy, yp, and rr arguments are of
109 * type N_Vector. The user_data parameter is the pointer user_data
110 * passed by the user to the IDASetRdata routine. This user-supplied
111 * pointer is passed to the user's res function every time it is called,
112 * to provide access in res to user data.
113 *
114 * A IDAResFn res should return a value of 0 if successful, a positive
115 * value if a recoverable error occured (e.g. yy has an illegal value),
116 * or a negative value if a nonrecoverable error occured. In the latter
117 * case, the program halts. If a recoverable error occured, the integrator
118 * will attempt to correct and retry.
119 * ----------------------------------------------------------------
120 */
121
122typedef int (*IDAResFn)(realtype tt, N_Vector yy, N_Vector yp,
123 N_Vector rr, void *user_data);
124
125/*
126 * -----------------------------------------------------------------
127 * Type : IDARootFn
128 * -----------------------------------------------------------------
129 * A function g, which defines a set of functions g_i(t,y,y') whose
130 * roots are sought during the integration, must have type IDARootFn.
131 * The function g takes as input the independent variable value t,
132 * the dependent variable vector y, and its t-derivative yp (= y').
133 * It stores the nrtfn values g_i(t,y,y') in the realtype array gout.
134 * (Allocation of memory for gout is handled within IDA.)
135 * The user_data parameter is the same as that passed by the user
136 * to the IDASetRdata routine. This user-supplied pointer is
137 * passed to the user's g function every time it is called.
138 *
139 * An IDARootFn should return 0 if successful or a non-zero value
140 * if an error occured (in which case the integration will be halted).
141 * -----------------------------------------------------------------
142 */
143
144typedef int (*IDARootFn)(realtype t, N_Vector y, N_Vector yp,
145 realtype *gout, void *user_data);
146
147/*
148 * -----------------------------------------------------------------
149 * Type : IDAEwtFn
150 * -----------------------------------------------------------------
151 * A function e, which sets the error weight vector ewt, must have
152 * type IDAEwtFn.
153 * The function e takes as input the current dependent variable y.
154 * It must set the vector of error weights used in the WRMS norm:
155 *
156 * ||y||_WRMS = sqrt [ 1/N * sum ( ewt_i * y_i)^2 ]
157 *
158 * Typically, the vector ewt has components:
159 *
160 * ewt_i = 1 / (reltol * |y_i| + abstol_i)
161 *
162 * The user_data parameter is the same as that passed by the user
163 * to the IDASetRdata routine. This user-supplied pointer is
164 * passed to the user's e function every time it is called.
165 * An IDAEwtFn e must return 0 if the error weight vector has been
166 * successfuly set and a non-zero value otherwise.
167 * -----------------------------------------------------------------
168 */
169
170typedef int (*IDAEwtFn)(N_Vector y, N_Vector ewt, void *user_data);

386/*
387 * ----------------------------------------------------------------
388 * Function : IDAReInit
389 * ----------------------------------------------------------------
390 * IDAReInit re-initializes IDA for the solution of a problem,
391 * where a prior call to IDAInit has been made.
392 * IDAReInit performs the same input checking and initializations
393 * that IDAInit does.
394 * But it does no memory allocation, assuming that the existing
395 * internal memory is sufficient for the new problem.
396 *
397 * The use of IDAReInit requires that the maximum method order,
398 * maxord, is no larger for the new problem than for the problem
399 * specified in the last call to IDAInit. This condition is
400 * automatically fulfilled if the default value for maxord is
401 * specified.
402 *
403 * Following the call to IDAReInit, a call to the linear solver
404 * specification routine is necessary if a different linear solver
405 * is chosen, but may not be otherwise. If the same linear solver
406 * is chosen, and there are no changes in its input parameters,
407 * then no call to that routine is needed.
408 *
409 * The first argument to IDAReInit is:
410 *
411 * ida_mem = pointer to IDA memory returned by IDACreate.
412 *
413 * All the remaining arguments to IDAReInit have names and
414 * meanings identical to those of IDAInit.
415 *
416 * The return value of IDAReInit is equal to SUCCESS = 0 if there
417 * were no errors; otherwise it is a negative int equal to:
418 * IDA_MEM_NULL indicating ida_mem was NULL, or
419 * IDA_NO_MALLOC indicating that ida_mem was not allocated.
420 * IDA_ILL_INPUT indicating an input argument was illegal
421 * (including an attempt to increase maxord).
422 * In case of an error return, an error message is also printed.
423 * ----------------------------------------------------------------
424 */
425
426SUNDIALS_EXPORT int IDAReInit(void *ida_mem,
427 realtype t0, N_Vector yy0, N_Vector yp0);

429/*
430 * -----------------------------------------------------------------
431 * Functions : IDASStolerances
432 * IDASVtolerances
433 * IDAWFtolerances
434 * -----------------------------------------------------------------
435 *
436 * These functions specify the integration tolerances. One of them
437 * MUST be called before the first call to IDA.
438 *
439 * IDASStolerances specifies scalar relative and absolute tolerances.
440 * IDASVtolerances specifies scalar relative tolerance and a vector
441 * absolute tolerance (a potentially different absolute tolerance
442 * for each vector component).
443 * IDAWFtolerances specifies a user-provides function (of type IDAEwtFn)
444 * which will be called to set the error weight vector.
445 *
446 * The tolerances reltol and abstol define a vector of error weights,
447 * ewt, with components
448 * ewt[i] = 1/(reltol*abs(y[i]) + abstol) (in the SS case), or
449 * ewt[i] = 1/(reltol*abs(y[i]) + abstol[i]) (in the SV case).
450 * This vector is used in all error and convergence tests, which
451 * use a weighted RMS norm on all error-like vectors v:
452 * WRMSnorm(v) = sqrt( (1/N) sum(i=1..N) (v[i]*ewt[i])^2 ),
453 * where N is the problem dimension.
454 *
455 * The return value of these functions is equal to IDA_SUCCESS = 0 if
456 * there were no errors; otherwise it is a negative int equal to:
457 * IDa_MEM_NULL indicating ida_mem was NULL (i.e.,
458 * IDACreate has not been called).
459 * IDA_NO_MALLOC indicating that ida_mem has not been
460 * allocated (i.e., IDAInit has not been
461 * called).
462 * IDA_ILL_INPUT indicating an input argument was illegal
463 * (e.g. a negative tolerance)
464 * In case of an error return, an error message is also printed.
465 * -----------------------------------------------------------------
466 */
467
468SUNDIALS_EXPORT int IDASStolerances(void *ida_mem, realtype reltol, realtype abstol);
469SUNDIALS_EXPORT int IDASVtolerances(void *ida_mem, realtype reltol, N_Vector abstol);
470SUNDIALS_EXPORT int IDAWFtolerances(void *ida_mem, IDAEwtFn efun);

566/*
567 * ----------------------------------------------------------------
568 * Function : IDACalcIC
569 * ----------------------------------------------------------------
570 * IDACalcIC calculates corrected initial conditions for the DAE
571 * system for a class of index-one problems of semi-implicit form.
572 * It uses Newton iteration combined with a Linesearch algorithm.
573 * Calling IDACalcIC is optional. It is only necessary when the
574 * initial conditions do not solve the given system. I.e., if
575 * y0 and yp0 are known to satisfy F(t0, y0, yp0) = 0, then
576 * a call to IDACalcIC is NOT necessary (for index-one problems).
577 *
578 * A call to IDACalcIC must be preceded by a successful call to
579 * IDAInit or IDAReInit for the given DAE problem, and by a
580 * successful call to the linear system solver specification
581 * routine.
582 *
583 * The call to IDACalcIC should precede the call(s) to IDASolve
584 * for the given problem.
585 *
586 * The arguments to IDACalcIC are as follows:
587 *
588 * ida_mem is the pointer to IDA memory returned by IDACreate.
589 *
590 * icopt is the option of IDACalcIC to be used.
591 * icopt = IDA_YA_YDP_INIT directs IDACalcIC to compute
592 * the algebraic components of y and differential
593 * components of y', given the differential
594 * components of y. This option requires that the
595 * N_Vector id was set through a call to IDASetId
596 * specifying the differential and algebraic
597 * components.
598 * icopt = IDA_Y_INIT directs IDACalcIC to compute all
599 * components of y, given y'. id is not required.
600 *
601 * tout1 is the first value of t at which a soluton will be
602 * requested (from IDASolve). (This is needed here to
603 * determine the direction of integration and rough scale
604 * in the independent variable t.)
605 *
606 *
607 * IDACalcIC returns an int flag. Its symbolic values and their
608 * meanings are as follows. (The numerical return values are set
609 * above in this file.) All unsuccessful returns give a negative
610 * return value. If IFACalcIC failed, y0 and yp0 contain
611 * (possibly) altered values, computed during the attempt.
612 *
613 * IDA_SUCCESS IDACalcIC was successful. The corrected
614 * initial value vectors were stored internally.
615 *
616 * IDA_MEM_NULL The argument ida_mem was NULL.
617 *
618 * IDA_ILL_INPUT One of the input arguments was illegal.
619 * See printed message.
620 *
621 * IDA_LINIT_FAIL The linear solver's init routine failed.
622 *
623 * IDA_BAD_EWT Some component of the error weight vector
624 * is zero (illegal), either for the input
625 * value of y0 or a corrected value.
626 *
627 * IDA_RES_FAIL The user's residual routine returned
628 * a non-recoverable error flag.
629 *
630 * IDA_FIRST_RES_FAIL The user's residual routine returned
631 * a recoverable error flag on the first call,
632 * but IDACalcIC was unable to recover.
633 *
634 * IDA_LSETUP_FAIL The linear solver's setup routine had a
635 * non-recoverable error.
636 *
637 * IDA_LSOLVE_FAIL The linear solver's solve routine had a
638 * non-recoverable error.
639 *
640 * IDA_NO_RECOVERY The user's residual routine, or the linear
641 * solver's setup or solve routine had a
642 * recoverable error, but IDACalcIC was
643 * unable to recover.
644 *
645 * IDA_CONSTR_FAIL IDACalcIC was unable to find a solution
646 * satisfying the inequality constraints.
647 *
648 * IDA_LINESEARCH_FAIL The Linesearch algorithm failed to find a
649 * solution with a step larger than steptol
650 * in weighted RMS norm.
651 *
652 * IDA_CONV_FAIL IDACalcIC failed to get convergence of the
653 * Newton iterations.
654 *
655 * ----------------------------------------------------------------
656 */
657
658SUNDIALS_EXPORT int IDACalcIC(void *ida_mem, int icopt, realtype tout1);

660/*
661 * ----------------------------------------------------------------
662 * Function : IDASolve
663 * ----------------------------------------------------------------
664 * IDASolve integrates the DAE over an interval in t, the
665 * independent variable. If itask is IDA_NORMAL, then the solver
666 * integrates from its current internal t value to a point at or
667 * beyond tout, then interpolates to t = tout and returns y(tret)
668 * in the user-allocated vector yret. In general, tret = tout.
669 * If itask is IDA_ONE_STEP, then the solver takes one internal
670 * step of the independent variable and returns in yret the value
671 * of y at the new internal independent variable value. In this
672 * case, tout is used only during the first call to IDASolve to
673 * determine the direction of integration and the rough scale of
674 * the problem. If tstop is enabled (through a call to IDASetStopTime),
675 * then IDASolve returns the solution at tstop. Once the integrator
676 * returns at a tstop time, any future testing for tstop is disabled
677 * (and can be reenabled only though a new call to IDASetStopTime).
678 * The time reached by the solver is placed in (*tret). The
679 * user is responsible for allocating the memory for this value.
680 *
681 * ida_mem is the pointer (void) to IDA memory returned by
682 * IDACreate.
683 *
684 * tout is the next independent variable value at which a
685 * computed solution is desired.
686 *
687 * tret is a pointer to a real location. IDASolve sets (*tret)
688 * to the actual t value reached, corresponding to the
689 * solution vector yret. In IDA_NORMAL mode, with no
690 * errors and no roots found, (*tret) = tout.
691 *
692 * yret is the computed solution vector. With no errors,
693 * yret = y(tret).
694 *
695 * ypret is the derivative of the computed solution at t = tret.
696 *
697 * Note: yret and ypret may be the same N_Vectors as y0 and yp0
698 * in the call to IDAInit or IDAReInit.
699 *
700 * itask is IDA_NORMAL or IDA_ONE_STEP. These two modes are described above.
701 *
702 *
703 * The return values for IDASolve are described below.
704 * (The numerical return values are defined above in this file.)
705 * All unsuccessful returns give a negative return value.
706 *
707 * IDA_SUCCESS
708 * IDASolve succeeded and no roots were found.
709 *
710 * IDA_ROOT_RETURN: IDASolve succeeded, and found one or more roots.
711 * If nrtfn > 1, call IDAGetRootInfo to see which g_i were found
712 * to have a root at (*tret).
713 *
714 * IDA_TSTOP_RETURN:
715 * IDASolve returns computed results for the independent variable
716 * value tstop. That is, tstop was reached.
717 *
718 * IDA_MEM_NULL:
719 * The IDA_mem argument was NULL.
720 *
721 * IDA_ILL_INPUT:
722 * One of the inputs to IDASolve is illegal. This includes the
723 * situation when a component of the error weight vectors
724 * becomes < 0 during internal stepping. It also includes the
725 * situation where a root of one of the root functions was found
726 * both at t0 and very near t0. The ILL_INPUT flag
727 * will also be returned if the linear solver function IDA---
728 * (called by the user after calling IDACreate) failed to set one
729 * of the linear solver-related fields in ida_mem or if the linear
730 * solver's init routine failed. In any case, the user should see
731 * the printed error message for more details.
732 *
733 * IDA_TOO_MUCH_WORK:
734 * The solver took mxstep internal steps but could not reach tout.
735 * The default value for mxstep is MXSTEP_DEFAULT = 500.
736 *
737 * IDA_TOO_MUCH_ACC:
738 * The solver could not satisfy the accuracy demanded by the user
739 * for some internal step.
740 *
741 * IDA_ERR_FAIL:
742 * Error test failures occurred too many times (=MXETF = 10) during
743 * one internal step.
744 *
745 * IDA_CONV_FAIL:
746 * Convergence test failures occurred too many times (= MXNCF = 10)
747 * during one internal step.
748 *
749 * IDA_LSETUP_FAIL:
750 * The linear solver's setup routine failed
751 * in an unrecoverable manner.
752 *
753 * IDA_LSOLVE_FAIL:
754 * The linear solver's solve routine failed
755 * in an unrecoverable manner.
756 *
757 * IDA_CONSTR_FAIL:
758 * The inequality constraints were violated,
759 * and the solver was unable to recover.
760 *
761 * IDA_REP_RES_ERR:
762 * The user's residual function repeatedly returned a recoverable
763 * error flag, but the solver was unable to recover.
764 *
765 * IDA_RES_FAIL:
766 * The user's residual function returned a nonrecoverable error
767 * flag.
768 *
769 * ----------------------------------------------------------------
770 */
