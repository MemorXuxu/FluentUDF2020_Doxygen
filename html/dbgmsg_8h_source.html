95/* Print an entire real vector in one line, possibly preceeded by the variable name.
96 * Save on the digits printed for legibility!
97 *
98 * tof==TrueOrFalse to print or not to print the variable name.
99 */
100#define dm_rvecmsg(n, V, tof) \
101{ \
102 int _ii; \
103 if (tof) dm_print2("%s: ",#V); \
104 for(_ii=0; _ii<n; _ii++) \
105 dm_print2_buf("%8.5e ",V[_ii]); \
106 if (tof) nlmsg; \
107 dm_flush; \
108}

136#define dm_rrvecmsg(n, V, tof) \

149/* Print an entire real matrix, line by line, possibly preceeded by the variable name.
150 * Save on the digits printed for legibility!
151 *
152 * tof==TrueOrFalse to print or not to print the variable name.
153 */
154#define dm_rmatmsg(m, n, M, tof) \
155{ \
156 int _ii,_jj; \
157 \
158 if (tof) dm_print2("%s: ",#M); \
159 if (tof) nlmsg; \
160 for(_ii=0; _ii<m; _ii++) \
161 { \
162 /* real *M##_ii=&M[_ii*n]; */ \
163 for(_jj=0; _jj<n; _jj++) \
164 { \
165 dm_print2_buf("%8.2g ",M[_ii][_jj]); \
166 if ((n-1) == _jj) nlmsg; \
167 } \
168 } \
169 if (tof) nlmsg; \
170 dm_flush; \
171}
