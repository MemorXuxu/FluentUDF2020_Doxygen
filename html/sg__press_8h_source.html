183#define BLEND_FACTOR_TIME(b_factor,phi,phi_M1,phi_M2, beta)\
184{ if(!tvd_time_bounded_second_order || ABS(phi) < 1.e-15)\
185 b_factor = 1.0; \
186 else \
187 {\
188 real phi_diff = ABS((phi - phi_M1)/phi); \
189 if(phi_diff < 1.e-10) \
190 b_factor = 1.0; \
191 else \
192 {\
193 real phi_r = (phi_M1 - phi_M2)/(phi - phi_M1); \
194 phi_r = fabs(phi_r); \
195 if(phi_r< 3.0) \
196 b_factor = 1.0;\
197 else \
198 b_factor = beta/(phi_r - 1.0); \
199 }\
200 }\
201}
