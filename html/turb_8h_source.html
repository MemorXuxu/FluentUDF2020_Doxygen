524#define COMPUTE_RANS_TURB_INTERIOR(crans,trans,l_scale,t_scale,k,eps)\
525{\
526 switch (M_turb_model)\
527 {\
528 case STD_KE:\
529 case RNG_KE:\
530 case REAL_KE:\
531 case LOWRE_KE:\
532 case EASM_KE:\
533 case DES_RKE:\
534 case V2F:\
535 {\
536 k = C_K(crans,trans);\
537 eps = C_D(crans,trans);\
538 l_scale = pow(k,3./2.)/eps;\
539 t_scale = k/eps;\
540 }\
541 break;\
542 case RSM_SKE:\
543 case RSM_RKE:\
544 {\
545 k = 0.5*(C_RUU(crans,trans)+C_RVV(crans,trans)+C_RWW(crans,trans));\
546 eps = C_D(crans,trans);\
547 l_scale = pow(k,3./2.)/eps;\
548 t_scale = k/eps;\
549 }\
550 break;\
551 case RSM_K_OMEGA:\
552 case RSM_BSL:\
553 {\
554 k = 0.5*(C_RUU(crans,trans)+C_RVV(crans,trans)+C_RWW(crans,trans));\
555 eps = GVAR_TURB(coeff, kw_beta_star_inf)*k*C_O(crans,trans);\
556 l_scale = pow(k,3./2.)/eps;\
557 t_scale = k/eps;\
558 }\
559 break;\
560 case TRANS_K_OMEGA:\
561 {\
562 k = C_K(crans,trans);\
563 eps = k*C_O(crans,trans);\
564 l_scale = pow(k,3./2.)/eps;\
565 t_scale = k/eps;\
566 }\
567 break;\
568 case K_OMEGA:\
569 case K_OMEGA_EASM:\
570 case WJ_BSL_EARSM:\
571 case BSL:\
572 case GEKO:\
573 case SST:\
574 case TRANS_SST:\
575 case SAS_SST:\
