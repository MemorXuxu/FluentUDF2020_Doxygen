77template<typename T>
78class MessageArgHelper
79{
80public:
81 typedef int (*ArgWriterFuncPtr)(RpcHandle channel, const T* value, int count);
82 typedef int (*ArgReaderFuncPtr)(RpcHandle channel, T* value, int count);
83
84 static ArgWriterFuncPtr ArgWriterFunc(ArgWriterFuncPtr func=NULL)
85 {
86 static ArgWriterFuncPtr writerFunc = NULL;
87 if (writerFunc == NULL)
88 {
89 if (func != NULL)
90 writerFunc = func;
91 else
92 throw IllegalArgTypeException(typeid(T).name());
93 }
94 return writerFunc;
95 }
96
97 static ArgReaderFuncPtr ArgReaderFunc(ArgReaderFuncPtr func=NULL)
98 {
99 static ArgReaderFuncPtr readerFunc = NULL;
100 if (readerFunc == NULL)
101 {
102 if (func != NULL)
103 readerFunc = func;
104 else
105 throw IllegalArgTypeException(typeid(T).name());
106 }
107 return readerFunc;
108 }
109
110 static RpcDataType DataType(RpcDataType* type=NULL)
111 {
112 static RpcDataType* dataType = NULL;
113 if (dataType == NULL)
114 {
115 if (type != NULL)
116 {
117 dataType = new RpcDataType();
118 *dataType = *type;
119 }
120 else
121 throw IllegalArgTypeException(typeid(T).name());
122 }
123 return *dataType;
124 }

145

152 }

173template<typename T>
174class MessageArrayArg : public MessageArgBase
175{
176public:
177 MessageArrayArg() : MessageArgBase(MessageArgHelper<T*>::DataType())
178 {
179 }
180
181 void AddToRpcMessageArgs(RpcMessageArgs* args) const
182 {
183 void* data = NULL;
184 int transferOwner = 0;
185
186 if (Value.size() > 0)
187 {
188 if (GetDataType() == kRpcDataTypeStringArray)
189 {
190 char** strArray = (char**)malloc(sizeof(char*)*Value.size());
191
192 for (unsigned int i = 0; i < Value.size(); i++)
193 {
194 std::string& s = (std::string&)Value[i];
195#if defined(NT_SYS)
196 strArray[i] = _strdup(s.c_str());
197#else
198 strArray[i] = strdup(s.c_str());
199#endif
200 }
201
202 data = strArray;
203 transferOwner = 1;
204 }
205 else
206 data = (void*)&Value[0];
207 }
208
209 RpcAddMessageArrayArg(args, GetDataType(), data, (RpcUInt)Value.size(), transferOwner);
210 }
211
212 void Read(RpcHandle channel)
213 {
214 int count;
215 CommunicationException::CheckForError(ansipc_ReadInt4(channel, &count, 1));
216
217 if (count > 0)
218 {
219 Value.resize(count);
220 if (GetDataType() == kRpcDataTypeStringArray)
221 {
222 std::vector<std::string>& strVec = (std::vector<std::string>&)Value;
223 for (int i = 0; i < count; i++)
224 {
225 char* str;
226 CommunicationException::CheckForError(ansipc_ReadStringExt(channel, &str));
227 strVec[i] = str;
228 free(str);
229 }
230 }
231 else
232 {
233 CommunicationException::CheckForError(
234 MessageArgHelper<T>::ArgReaderFunc()(channel, &Value[0], count));
235 }
236 }
237 }
238
239 void Write(RpcHandle channel) const
240 {
241 WriteDataType(channel);
242
243 int count = (int)Value.size();
244 CommunicationException::CheckForError(ansipc_WriteInt4(channel, &count, 1));
245
246 if (Value.size() > 0)
247 {
248 if (GetDataType() == kRpcDataTypeStringArray)
249 {
250 for (size_t i = 0; i < Value.size(); i++)
251 {
252 std::string& s = (std::string&)Value[i];
253 CommunicationException::CheckForError(ansipc_WriteString(channel, (char*)s.c_str()));
254 }
255 }
256 else
257 {
258 CommunicationException::CheckForError(
259 MessageArgHelper<T>::ArgWriterFunc()(channel, &Value[0], (int)Value.size()));
260 }
261 }
262 }
263
264 std::vector<T> Value;
265};

356 static MessageArgBase* CreateArg(RpcDataType dataType)
357 {
358 ArgCreatorMap& creatorMap = GetCreatorMap();
359 ArgCreatorMapIterator it = creatorMap.find(dataType);
360 if (it != creatorMap.end())
361 {
362 return (*it).second->CreateArg();
363 }
364 else
365 {
366 std::ostringstream oss;
367 oss << "Invalid message arg data type: " << dataType;
368 throw RpcException(oss.str());
369 }
370 }
