36/*
37 * Given the operands for calling a ufunc, should determine the
38 * calculation input and output data types and return an inner loop function.
39 * This function should validate that the casting rule is being followed,
40 * and fail if it is not.
41 *
42 * For backwards compatibility, the regular type resolution function does not
43 * support auxiliary data with object semantics. The type resolution call
44 * which returns a masked generic function returns a standard NpyAuxData
45 * object, for which the NPY_AUXDATA_FREE and NPY_AUXDATA_CLONE macros
46 * work.
47 *
48 * ufunc: The ufunc object.
49 * casting: The 'casting' parameter provided to the ufunc.
50 * operands: An array of length (ufunc->nin + ufunc->nout),
51 * with the output parameters possibly NULL.
52 * type_tup: Either NULL, or the type_tup passed to the ufunc.
53 * out_dtypes: An array which should be populated with new
54 * references to (ufunc->nin + ufunc->nout) new
55 * dtypes, one for each input and output. These
56 * dtypes should all be in native-endian format.
57 *
58 * Should return 0 on success, -1 on failure (with exception set),
59 * or -2 if Py_NotImplemented should be returned.
60 */
61typedef int (PyUFunc_TypeResolutionFunc)(
62 struct _tagPyUFuncObject *ufunc,
63 NPY_CASTING casting,
64 PyArrayObject **operands,
65 PyObject *type_tup,
66 PyArray_Descr **out_dtypes);

68/*
69 * Given an array of DTypes as returned by the PyUFunc_TypeResolutionFunc,
70 * and an array of fixed strides (the array will contain NPY_MAX_INTP for
71 * strides which are not necessarily fixed), returns an inner loop
72 * with associated auxiliary data.
73 *
74 * For backwards compatibility, there is a variant of the inner loop
75 * selection which returns an inner loop irrespective of the strides,
76 * and with a void* static auxiliary data instead of an NpyAuxData *
77 * dynamically allocatable auxiliary data.
78 *
79 * ufunc: The ufunc object.
80 * dtypes: An array which has been populated with dtypes,
81 * in most cases by the type resolution function
82 * for the same ufunc.
83 * fixed_strides: For each input/output, either the stride that
84 * will be used every time the function is called
85 * or NPY_MAX_INTP if the stride might change or
86 * is not known ahead of time. The loop selection
87 * function may use this stride to pick inner loops
88 * which are optimized for contiguous or 0-stride
89 * cases.
90 * out_innerloop: Should be populated with the correct ufunc inner
91 * loop for the given type.
92 * out_innerloopdata: Should be populated with the void* data to
93 * be passed into the out_innerloop function.
94 * out_needs_api: If the inner loop needs to use the Python API,
95 * should set the to 1, otherwise should leave
96 * this untouched.
97 */

99 struct _tagPyUFuncObject *ufunc,
100 PyArray_Descr **dtypes,
101 PyUFuncGenericFunction *out_innerloop,
102 void **out_innerloopdata,
103 int *out_needs_api);
