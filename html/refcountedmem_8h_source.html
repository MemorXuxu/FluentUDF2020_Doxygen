13/* Use Alloc_Ref_Counted_Memory to allocate memory. The user needs to
14 * provide some owner for the memory being allocated. When the owner is
15 * copied, Ref_Counted_Memory_Copied should be called. This will
16 * increase the reference counts for all memories that the owner (src)
17 * holds.
18 e.g: Lets say we have a container
19 typedef struct bag_struct {
20 char *name;
21 char *values;
22 } Bag;
23
24 // We want to allow shallow copy as well as deep copy of Bag.
25 // The following function does it one by one.
26
27 void func()
28 {
29 Bag A;
30 A.name = Alloc_Ref_Counted_Memory(&A, 10); // owner is A and size 10
31 A.values = Alloc_Ref_Counted_Memory(&A, 100);
32
33 Bag B = A; // shallow copy A into B.
34 // inform the system that A is now copied to B
35 Ref_Counted_Memory_Copied(&A, &B);
36
37 // Now let us create different memory into B.
38 Free_Ref_Counted_Memory(&B, B.name);
39 Free_Ref_Counted_Memory(&B, B.values);
40 B.name = Alloc_Ref_Counted_Memory(&B, 15); // Now B gets its own memory
41 B.values = Alloc_Ref_Counted_Memory(&B, 20); // B values are its own too.
42
43 // Before A & B go out of context. Inform the system to free up the
44 // memories
45 Free_Ref_Counted_Memory(&A, NULL); // all memory contained in A should be freed
46 Free_Ref_Counted_Memory(&B, NULL); // all memory contained in B should be freed
47 }
48 */
49FLUENT_EXPORT void *Alloc_Ref_Counted_Memory(void *owner, size_t size);
50FLUENT_EXPORT void Ref_Counted_Memory_Copied(void *src, void *dest);
51/* If ptr is a ref counted memory, its ref count will be reduced.

53 If owner doesn't own any ref counted memory, this won't do anything.
54*/
