85/* parallel implementation of the N_Vector 'content' structure
86 contains the global and local lengths of the vector, a pointer
87 to an array of 'realtype components', the MPI communicator,
88 and a flag indicating ownership of the data */

92 long int global_length; /* global vector length */
93 booleantype own_data; /* ownership of data */

95 MPI_Comm comm; /* pointer to MPI communicator */
96};

100/*
101 * -----------------------------------------------------------------
102 * PART II: macros NV_CONTENT_P, NV_DATA_P, NV_OWN_DATA_P,
103 * NV_LOCLENGTH_P, NV_GLOBLENGTH_P,NV_COMM_P, and NV_Ith_P
104 * -----------------------------------------------------------------
105 * In the descriptions below, the following user declarations
106 * are assumed:
107 *
108 * N_Vector v;
109 * long int v_len, s_len, i;
110 *
111 * (1) NV_CONTENT_P
112 *
113 * This routines gives access to the contents of the parallel
114 * vector N_Vector.
115 *
116 * The assignment v_cont = NV_CONTENT_P(v) sets v_cont to be
117 * a pointer to the parallel N_Vector content structure.
118 *
119 * (2) NV_DATA_P, NV_OWN_DATA_P, NV_LOCLENGTH_P, NV_GLOBLENGTH_P,
120 * and NV_COMM_P
121 *
122 * These routines give access to the individual parts of
123 * the content structure of a parallel N_Vector.
124 *
125 * The assignment v_data = NV_DATA_P(v) sets v_data to be
126 * a pointer to the first component of the local data for
127 * the vector v. The assignment NV_DATA_P(v) = data_v sets
128 * the component array of v to be data_V by storing the
129 * pointer data_v.
130 *
131 * The assignment v_llen = NV_LOCLENGTH_P(v) sets v_llen to
132 * be the length of the local part of the vector v. The call
133 * NV_LOCLENGTH_P(v) = llen_v sets the local length
134 * of v to be llen_v.
135 *
136 * The assignment v_glen = NV_GLOBLENGTH_P(v) sets v_glen to
137 * be the global length of the vector v. The call
138 * NV_GLOBLENGTH_P(v) = glen_v sets the global length of v to
139 * be glen_v.
140 *
141 * The assignment v_comm = NV_COMM_P(v) sets v_comm to be the
142 * MPI communicator of the vector v. The assignment
143 * NV_COMM_C(v) = comm_v sets the MPI communicator of v to be
144 * comm_v.
145 *
146 * (3) NV_Ith_P
147 *
148 * In the following description, the components of the
149 * local part of an N_Vector are numbered 0..n-1, where n
150 * is the local length of (the local part of) v.
151 *
152 * The assignment r = NV_Ith_P(v,i) sets r to be the value
153 * of the ith component of the local part of the vector v.
154 * The assignment NV_Ith_P(v,i) = r sets the value of the
155 * ith local component of v to be r.
156 *
157 * Note: When looping over the components of an N_Vector v, it is
158 * more efficient to first obtain the component array via
159 * v_data = NV_DATA_P(v) and then access v_data[i] within the
160 * loop than it is to use NV_Ith_P(v,i) within the loop.
161 * -----------------------------------------------------------------
162 */
163
164#define NV_CONTENT_P(v) ( (N_VectorContent_Parallel)(v->content) )
165
166#define NV_LOCLENGTH_P(v) ( NV_CONTENT_P(v)->local_length )
167
168#define NV_GLOBLENGTH_P(v) ( NV_CONTENT_P(v)->global_length )
169
170#define NV_OWN_DATA_P(v) ( NV_CONTENT_P(v)->own_data )
171
172#define NV_DATA_P(v) ( NV_CONTENT_P(v)->data )
173
174#define NV_COMM_P(v) ( NV_CONTENT_P(v)->comm )
175
176#define NV_Ith_P(v,i) ( NV_DATA_P(v)[i] )
