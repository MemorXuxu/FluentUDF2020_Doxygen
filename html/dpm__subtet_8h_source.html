35 * @brief
36 * struct to store necessary information regarding the
37 * parent cell face with which the subtet is associated.
38 *
39 */
40typedef struct subtet_tracking_location_struct
41{
42 /* parent cell face that contains one side of the subtet */
43 face_t f;
44 Thread *tf;
45
46 /* all nodes of the associated face */
47 Node *faceNodes[MAX_FACE_NODES + 1];
48 int numFacePts; /* 3 for tri face, 4 for quad face, etc. */
49
50 /* the coordinates and velocity of the face nodes */
51 double facePts[MAX_FACE_NODES + 1][3];
52 double facePtsVel[MAX_FACE_NODES + 1][3];
53
54 /* nodes used for the subtet (beside face and cell centroids) */
55 Node *tetNodes[4];
56 int numTetNodes; /* 2 for poly face, 3 for quad face, 4 for tet cells, etc. */
57
58 /* coordinates and velocity of the subtet points */
59 double tetPts[4][3];
60 double tetPtsVel[4][3];
61
62 /* index of first face node when constructing subtet */
63 int inode;
64
65 /* forward (0) or reverse (1) ordering */
66 int order;
67
68 /* particle might have changed cell */
69 cxboolean eval_bc;
70
71 /* validity of subtet - indicates if this subtet is degenerate */
72 cxboolean degenerate;
73
74 /* hanging node parameters - use child face nodes if face, f, corresponds to a parent face */
75 int faceIndxChild;
76 int inodeChild;
77 face_t fChild;
78
79 cxboolean hold_constant; /* flag to prevent subtet from being modified */
80 cxboolean new_subtet; /* the particle has entered a new subtet */
81 cxboolean hold_vel_constant; /* hold the velocity constant for a timestep after a reflection */
82
83 double r_matrix[3][3];
84
85 int in_plane_index; /* subtet face index when traveling within a plane */
86
87
88 /* variables for wall film particles */
89 cxboolean wall_film_needs_update; /* indicates the particle has left the last face and is searching for the next */
90
91 /* wall film face */
92 int wall_film_face_index; /* subtet face index */
93 face_t f_wall_film; /* cell face index */
94 Thread *tf_wall_film; /* NNULLP(tf_wall_film) indicates the particle is on the wall */
95 face_t f_interface; /* face index for the nci */
96 Thread *tf_interface; /* face thread for the nci */
97
98 Node *edgeNodes[2]; /* pointers to the nodes of the wall-film edge crossed */
99 cxboolean on_edge; /* indicates we have just completed grid walking to the new wall film face */
100
101 double normal[3]; /* wall film face normal */
102
103 /* film variables used for physics, not tracking (and can therefore be reals) */
104 real film_vel[3]; /* wall film velocity */
105 real film_height; /* wall film height */
106 real film_face_area; /* area of the wall film face */
107
108 cxindex global_face_id; /* global face index of face crossed (for particle migration) */
109 cxboolean separates; /* indicates all separation criteria except face angle have been met */
110
111 Tracked_Particle *tp;
112
113} Subtet_Tracking_Location;
