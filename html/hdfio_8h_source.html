53typedef struct thread_data_manager_t__
54{
55 MPT_Datatype (*get_mpt_datatype)();
56
57 /* before any write operation */
58 /* NULL: not required */
59 int (*update_before_write)();
60
61 /* should return true if at least one element of this thread has data */
62 /* NULL: all threads (cell, face and node) will be checked for this data */
63 /* recommended to use if information is available */
64 cxboolean (*thread_has_data)(Thread *);
65
66 /* returns if elements in use of this thread has same amount of data */
67 /* NULL: no thread in use has uniform data */
68 /* suggested to use this if uniform data is used */
69 cxboolean (*thread_has_uniform_data)(Thread *, int *);
70
71 /* returns number of data points of this element to be written */
72 /* this will not be called for threads which have uniform data */
73 /* NULL: if all elements of all threads *in use* have uniform data */
74 int (*get_num_cell_data)(cell_t, Thread *);
75 int (*get_num_face_data)(face_t, Thread *);
76 int (*get_num_node_data)(Node *);
77
78 /* number of rows in this compute-process */
79 /* number of columns across all compute-processes; should be same */
80 int (*get_num_direct_data)(size_t *, int *);
81
82 /* gets the data in a void pointer */
83 /* at least one needs to be valid */
84 int (*get_cell_data)(cell_t, Thread *, void *);
85 int (*get_face_data)(face_t, Thread *, void *);
86 int (*get_node_data)(Node *, void *);
87
88 /* fill up data present in this compute-process */
89 int (*get_direct_data)(size_t, int, void *);
90
91 /* after all write operation */
92 /* NULL: not required */
93 int (*update_after_write)();
94
95 /* before any read operation */
96 /* NULL: not required */
97 int (*update_before_read)();
98
99 /* This gets called if data is not available, prepare your model that way */
100 /* NULL: not needed */
101 int (*data_not_available)();
102
103 /* This thread is read from file and is used to set this information */
104 /* NULL: not required */
105 int (*set_thread_has_data)(Thread *);
106
107 /* Allocate memory on active threads */
108 /* NULL: not required */
109 int (*update_before_reading_num_data)();
110
111 /* Pass information about uniform data from file to memory */
112 /* NULL: not required */
113 int (*set_thread_has_uniform_data)(Thread *, int);
114
115 /* Pass information to solver which is read from file */
116 /* NULL: not required */
117 int (*set_num_cell_data)(cell_t, Thread *, int);
118 int (*set_num_face_data)(face_t, Thread *, int);
119 int (*set_num_node_data)(Node *, int);
120
121 /* number of rows and columns in the file */
122 int (*set_num_direct_data)(size_t, int);
123
124 /* after filling the num-data information */
125 /* NULL: not required */
126 int (*update_after_reading_num_data)();
127
128 /* Allocate memory on active threads */
129 /* NULL: not required */
130 int (*update_before_reading_data)();
131
132 /* how many elements this process should read */
133 /* NULL: everything will be read in process-0 */
134 int (*get_num_direct_data_reading_local)(size_t *);
135
136 /* Set the data */
137 /* al least one needs to be valid */
138 int (*set_cell_data)(cell_t, Thread *, int, MPT_Datatype, void *);
139 int (*set_face_data)(face_t, Thread *, int, MPT_Datatype, void *);
140 int (*set_node_data)(Node *, int, MPT_Datatype, void *);
141
142 int (*set_direct_data)(size_t, int, MPT_Datatype, void *);
143
144 /* after filling the data */
145 /* NULL: not required */
146 int (*update_after_reading_data)();
147
148 /* after all read operation */
149 /* NULL: not required */
150 int (*update_after_read)();
151
152 /* set appropriate flag */
153 /* must use for direct data */
154 int flags;
