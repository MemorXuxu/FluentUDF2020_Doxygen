14 * This is the header file for the IDABBDPRE module, for a
15 * band-block-diagonal preconditioner, i.e. a block-diagonal
16 * matrix with banded blocks, for use with IDA and
17 * IDASPGMR/IDASPBCG/IDASPTFQMR.
18 *
19 * Summary:
20 *
21 * These routines provide a preconditioner matrix that is
22 * block-diagonal with banded blocks. The blocking corresponds
23 * to the distribution of the dependent variable vector y among
24 * the processors. Each preconditioner block is generated from
25 * the Jacobian of the local part (on the current processor) of a
26 * given function G(t,y,y') approximating F(t,y,y'). The blocks
27 * are generated by a difference quotient scheme on each processor
28 * independently. This scheme utilizes an assumed banded structure
29 * with given half-bandwidths, mudq and mldq. However, the banded
30 * Jacobian block kept by the scheme has half-bandwiths mukeep and
31 * mlkeep, which may be smaller.
32 *
33 * The user's calling program should have the following form:
34 *
35 * #include <ida/ida_bbdpre.h>
36 * #include <nvector_parallel.h>
37 * ...
38 * y0 = N_VNew_Parallel(...);
39 * yp0 = N_VNew_Parallel(...);
40 * ...
41 * ida_mem = IDACreate(...);
42 * ier = IDAInit(...);
43 * ...
44 * flag = IDASptfqmr(ida_mem, maxl);
45 * -or-
46 * flag = IDASpgmr(ida_mem, maxl);
47 * -or-
48 * flag = IDASpbcg(ida_mem, maxl);
49 * ...
50 * flag = IDABBDPrecInit(ida_mem, Nlocal, mudq, mldq,
51 * mukeep, mlkeep, dq_rel_yy, Gres, Gcomm);
52 * ...
53 * ier = IDASolve(...);
54 * ...
55 * IDAFree(&ida_mem);
56 *
57 * N_VDestroy(y0);
58 * N_VDestroy(yp0);
59 *
60 * The user-supplied routines required are:
61 *
62 * res is the function F(t,y,y') defining the DAE system to
63 * be solved: F(t,y,y') = 0.
64 *
65 * Gres is the function defining a local approximation
66 * G(t,y,y') to F, for the purposes of the preconditioner.
67 *
68 * Gcomm is the function performing communication needed
69 * for Glocal.
70 *
71 * Notes:
72 *
73 * 1) This header file is included by the user for the definition
74 * of the IBBDPrecData type and for needed function prototypes.
75 *
76 * 2) The IDABBDPrecInit call includes half-bandwidths mudq and
77 * mldq to be used in the approximate Jacobian. They need
78 * not be the true half-bandwidths of the Jacobian of the
79 * local block of G, when smaller values may provide a greater
80 * efficiency. Similarly, mukeep and mlkeep, specifying the
81 * bandwidth kept for the approximate Jacobian, need not be
82 * the true half-bandwidths. Also, mukeep, mlkeep, mudq, and
83 * mldq need not be the same on every processor.
84 *
85 * 3) The actual name of the user's res function is passed to
86 * IDAInit, and the names of the user's Gres and Gcomm
87 * functions are passed to IDABBDPrecInit.
88 *
89 * 4) The pointer to the user-defined data block user_data, which
90 * is set through IDASetUserData is also available to the user
91 * in glocal and gcomm.
92 *
93 * 5) Optional outputs specific to this module are available by
94 * way of routines listed below. These include work space sizes
95 * and the cumulative number of glocal calls. The costs
96 * associated with this module also include nsetups banded LU
97 * factorizations, nsetups gcomm calls, and nps banded
98 * backsolve calls, where nsetups and nps are integrator
99 * optional outputs.
100 * -----------------------------------------------------------------
101 */

112/*
113 * -----------------------------------------------------------------
114 * Type : IDABBDLocalFn
115 * -----------------------------------------------------------------
116 * The user must supply a function G(t,y,y') which approximates
117 * the function F for the system F(t,y,y') = 0, and which is
118 * computed locally (without interprocess communication).
119 * (The case where G is mathematically identical to F is allowed.)
120 * The implementation of this function must have type IDABBDLocalFn.
121 *
122 * This function takes as input the independent variable value tt,
123 * the current solution vector yy, the current solution
124 * derivative vector yp, and a pointer to the user-defined data
125 * block user_data. It is to compute the local part of G(t,y,y')
126 * and store it in the vector gval. (Providing memory for yy and
127 * gval is handled within this preconditioner module.) It is
128 * expected that this routine will save communicated data in work
129 * space defined by the user, and made available to the
130 * preconditioner function for the problem. The user_data
131 * parameter is the same as that passed by the user to the
132 * IDASetRdata routine.
133 *
134 * An IDABBDLocalFn Gres is to return an int, defined in the same
135 * way as for the residual function: 0 (success), +1 or -1 (fail).
136 * -----------------------------------------------------------------
137 */
138
139typedef int (*IDABBDLocalFn)(long int Nlocal, realtype tt,
140 N_Vector yy, N_Vector yp, N_Vector gval,
141 void *user_data);
142
143/*
144 * -----------------------------------------------------------------
145 * Type : IDABBDCommFn
146 * -----------------------------------------------------------------
147 * The user may supply a function of type IDABBDCommFn which
148 * performs all interprocess communication necessary to
149 * evaluate the approximate system function described above.
150 *
151 * This function takes as input the solution vectors yy and yp,
152 * and a pointer to the user-defined data block user_data. The
153 * user_data parameter is the same as that passed by the user to
154 * the IDASetUserData routine.
155 *
156 * The IDABBDCommFn Gcomm is expected to save communicated data in
157 * space defined with the structure *user_data.
158 *
159 * A IDABBDCommFn Gcomm returns an int value equal to 0 (success),
160 * > 0 (recoverable error), or < 0 (unrecoverable error).
161 *
162 * Each call to the IDABBDCommFn is preceded by a call to the system
163 * function res with the same vectors yy and yp. Thus the
164 * IDABBDCommFn gcomm can omit any communications done by res if
165 * relevant to the evaluation of the local function glocal.
166 * A NULL communication function can be passed to IDABBDPrecInit
167 * if all necessary communication was done by res.
168 * -----------------------------------------------------------------
169 */
170
171typedef int (*IDABBDCommFn)(long int Nlocal, realtype tt,
172 N_Vector yy, N_Vector yp,
173 void *user_data);

175/*
176 * -----------------------------------------------------------------
177 * Function : IDABBDPrecInit
178 * -----------------------------------------------------------------
179 * IDABBDPrecInit allocates and initializes the BBD preconditioner.
180 *
181 * The parameters of IDABBDPrecInit are as follows:
182 *
183 * ida_mem is a pointer to the memory blockreturned by IDACreate.
184 *
185 * Nlocal is the length of the local block of the vectors yy etc.
186 * on the current processor.
187 *
188 * mudq, mldq are the upper and lower half-bandwidths to be used
189 * in the computation of the local Jacobian blocks.
190 *
191 * mukeep, mlkeep are the upper and lower half-bandwidths to be
192 * used in saving the Jacobian elements in the local
193 * block of the preconditioner matrix PP.
194 *
195 * dq_rel_yy is an optional input. It is the relative increment
196 * to be used in the difference quotient routine for
197 * Jacobian calculation in the preconditioner. The
198 * default is sqrt(unit roundoff), and specified by
199 * passing dq_rel_yy = 0.
200 *
201 * Gres is the name of the user-supplied function G(t,y,y')
202 * that approximates F and whose local Jacobian blocks
203 * are to form the preconditioner.
204 *
205 * Gcomm is the name of the user-defined function that performs
206 * necessary interprocess communication for the
207 * execution of glocal.
208 *
209 * The return value of IDABBDPrecInit is one of:
210 * IDASPILS_SUCCESS if no errors occurred
211 * IDASPILS_MEM_NULL if the integrator memory is NULL
212 * IDASPILS_LMEM_NULL if the linear solver memory is NULL
213 * IDASPILS_ILL_INPUT if an input has an illegal value
214 * IDASPILS_MEM_FAIL if a memory allocation request failed
215 * -----------------------------------------------------------------
216 */
217
