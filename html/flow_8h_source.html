282 (A[0][1]*B[1][0])+(A[0][2]*B[2][0])+\
283 (A[1][0]*B[0][1])+(A[1][2]*B[2][1])+\
284 (A[2][0]*B[0][2])+(A[2][1]*B[1][2]))
