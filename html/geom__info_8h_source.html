86#define GET_INTERIOR_FACE_GEOMETRY_LD(f,t,gi,fA,d0,d1,A_by_dr0,A_by_dr1)\
87{\
88 fA = NV_MAG2(gi.A[f]); /* face Area magnitude */ \
89 A_by_dr0 = fA/NV_DOT(gi.dr0[f],gi.A[f]); \
90 A_by_dr1 = fA/NV_DOT(gi.dr1[f],gi.A[f]); \
91 fA = sqrt(fA); \
92 d0 = NV_DOT(gi.dr0[f],gi.A[f])/fA; /* dr0 = X_f - X_c0 */ \
93 d1 = NV_DOT(gi.dr1[f],gi.A[f])/fA; /* dr1 = X_f - X_c1 */ \
94 if(d0 < REAL_MIN) d0 = ABS(d0); /* in case of dr0 or dr1 */ \
95 if(d1 < REAL_MIN) d1 = ABS(d1); /* in opp. dir. w.r.t A */ \
96}
97#define GET_INTERIOR_FACE_GEOMETRY_LONG(f,t,gi,es0,es1,A_by_es0,A_by_es1) \
98{ \
99 real Amag2 = NV_MAG2(gi.A[f]); \
100 NV_VS(es0, =, gi.dr0[f], /, NV_MAG(gi.dr0[f])); \
101 NV_VS(es1, =, gi.dr1[f], /, NV_MAG(gi.dr1[f])); \
102 A_by_es0 = Amag2/NV_DOT(es0,gi.A[f]); \
103 A_by_es1 = Amag2/NV_DOT(es1,gi.A[f]); \
