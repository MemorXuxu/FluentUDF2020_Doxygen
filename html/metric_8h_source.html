144# define Calc_F_Area(A,f,t)\
145 switch (F_NNODES(f,t))\
146 {\
147 case MIXED_FACE: /* Accumulated face */\
148 NV_V (A, =, ACCUMULATED_F_AREA(f,t));\
149 break;\
150\
151 case LINEAR_FACE: /* 2D face */\
152 default: \
153 if (rp_axi)\
154 {\
155 A[0] = AXI_AREA_X(F_NODE(f,t,0),F_NODE(f,t,1));\
156 A[1] = AXI_AREA_Y(F_NODE(f,t,0),F_NODE(f,t,1));\
157 }\
158 else\
159 {\
160 A[0] = TWOD_AREA_X(F_NODE(f,t,0),F_NODE(f,t,1));\
161 A[1] = TWOD_AREA_Y(F_NODE(f,t,0),F_NODE(f,t,1));\
162 }\
163 break;\
164 }
165
166# else /* RP_3D */
167
168# define Calc_F_Area(A,f,t)\
169 switch (F_NNODES(f,t))\
170 {\
171 case MIXED_FACE: /* Accumulated face */\
172 NV_V (A, =, ACCUMULATED_F_AREA(f,t));\
173 break;\
174\
175 case TRI_FACE: /* 3D triangular face */\
176 {\
177 Node *n0 = F_NODE(f,t,0);\
178 Node *n1 = F_NODE(f,t,1);\
179 Node *n2 = F_NODE(f,t,2);\
180 A[0] = TRI_AREA_X(n0,n1,n2);\
181 A[1] = TRI_AREA_Y(n0,n1,n2);\
182 A[2] = TRI_AREA_Z(n0,n1,n2);\
183 }\
184 break;\
185\
186 case QUAD_FACE: /* 3D quad face */\
187 {\
188 Node *n0 = F_NODE(f,t,0);\
189 Node *n1 = F_NODE(f,t,1);\
190 Node *n2 = F_NODE(f,t,2);\
191 Node *n3 = F_NODE(f,t,3);\
192 A[0] = QUAD_AREA_X(n0,n1,n2,n3);\
193 A[1] = QUAD_AREA_Y(n0,n1,n2,n3);\
194 A[2] = QUAD_AREA_Z(n0,n1,n2,n3);\
195 }\
196 break;\
197\
198 case POLY_FACE:\
199 default: /* 3D polygonal face */\
200 {\
201 register int _i;\
202 \
203 NV_S (A, =, 0.);\
204\
205 for (_i = 0; _i < F_NNODES(f,t); ++_i)\
206 {\
207 Node *n0 = F_NODE (f, t, _i);\
208 Node *n1 = F_NODE (f, t, (_i+1)%F_NNODES(f,t));\
209 ND_MAP(A[0],A[1],A[2], +=,\
210 (NODE_Y(n1) + NODE_Y(n0))*(NODE_Z(n1) - NODE_Z(n0)),\
211 (NODE_Z(n1) + NODE_Z(n0))*(NODE_X(n1) - NODE_X(n0)),\
212 (NODE_X(n1) + NODE_X(n0))*(NODE_Y(n1) - NODE_Y(n0)));\
213 }\
214 NV_S (A, /=, 2.);\
215 }\
216 break;\
217 }
218# endif /* RP_3D */
