41/*
42 * -----------------------------------------------------------------
43 * Type : IDASpilsPrecSetupFn
44 * -----------------------------------------------------------------
45 * The optional user-supplied functions PrecSetup and PrecSolve
46 * together must define the left preconditoner matrix P
47 * approximating the system Jacobian matrix
48 * J = dF/dy + c_j*dF/dy'
49 * (where the DAE system is F(t,y,y') = 0), and solve the linear
50 * systems P z = r. PrecSetup is to do any necessary setup
51 * operations, and PrecSolve is to compute the solution of
52 * P z = r.
53 *
54 * The preconditioner setup function PrecSetup is to evaluate and
55 * preprocess any Jacobian-related data needed by the
56 * preconditioner solve function PrecSolve. This might include
57 * forming a crude approximate Jacobian, and performing an LU
58 * factorization on it. This function will not be called in
59 * advance of every call to PrecSolve, but instead will be called
60 * only as often as necessary to achieve convergence within the
61 * Newton iteration. If the PrecSolve function needs no
62 * preparation, the PrecSetup function can be NULL.
63 *
64 * Each call to the PrecSetup function is preceded by a call to
65 * the system function res with the same (t,y,y') arguments.
66 * Thus the PrecSetup function can use any auxiliary data that is
67 * computed and saved by the res function and made accessible
68 * to PrecSetup.
69 *
70 * A preconditioner setup function PrecSetup must have the
71 * prototype given below. Its parameters are as follows:
72 *
73 * tt is the current value of the independent variable t.
74 *
75 * yy is the current value of the dependent variable vector,
76 * namely the predicted value of y(t).
77 *
78 * yp is the current value of the derivative vector y',
79 * namely the predicted value of y'(t).
80 *
81 * rr is the current value of the residual vector F(t,y,y').
82 *
83 * c_j is the scalar in the system Jacobian, proportional to 1/hh.
84 *
85 * user_data is a pointer to user data, the same as the user_data
86 * parameter passed to IDASetUserData.
87 *
88 * tmp1, tmp2, tmp3 are pointers to vectors of type N_Vector
89 * which can be used by an IDASpilsPrecSetupFn routine
90 * as temporary storage or work space.
91 *
92 * NOTE: If the user's preconditioner needs other quantities,
93 * they are accessible as follows: hcur (the current stepsize)
94 * and ewt (the error weight vector) are accessible through
95 * IDAGetCurrentStep and IDAGetErrWeights, respectively (see
96 * ida.h). The unit roundoff is available as
97 * UNIT_ROUNDOFF defined in sundials_types.h
98 *
99 * The IDASpilsPrecSetupFn should return
100 * 0 if successful,
101 * a positive int if a recoverable error occurred, or
102 * a negative int if a nonrecoverable error occurred.
103 * In the case of a recoverable error return, the integrator will
104 * attempt to recover by reducing the stepsize (which changes cj).
105 * -----------------------------------------------------------------
106 */
107
108typedef int (*IDASpilsPrecSetupFn)(realtype tt,
109 N_Vector yy, N_Vector yp, N_Vector rr,
110 realtype c_j, void *user_data,
111 N_Vector tmp1, N_Vector tmp2,
112 N_Vector tmp3);
113
114/*
115 * -----------------------------------------------------------------
116 * Type : IDASpilsPrecSolveFn
117 * -----------------------------------------------------------------
118 * The optional user-supplied function PrecSolve must compute a
119 * solution to the linear system P z = r, where P is the left
120 * preconditioner defined by the user. If no preconditioning
121 * is desired, pass NULL for PrecSolve to IDASp*.
122 *
123 * A preconditioner solve function PrecSolve must have the
124 * prototype given below. Its parameters are as follows:
125 *
126 * tt is the current value of the independent variable t.
127 *
128 * yy is the current value of the dependent variable vector y.
129 *
130 * yp is the current value of the derivative vector y'.
131 *
132 * rr is the current value of the residual vector F(t,y,y').
133 *
134 * rvec is the input right-hand side vector r.
135 *
136 * zvec is the computed solution vector z.
137 *
138 * c_j is the scalar in the system Jacobian, proportional to 1/hh.
139 *
140 * delta is an input tolerance for use by PrecSolve if it uses an
141 * iterative method in its solution. In that case, the
142 * the residual vector r - P z of the system should be
143 * made less than delta in weighted L2 norm, i.e.,
144 * sqrt [ Sum (Res[i]*ewt[i])^2 ] < delta .
145 * Note: the error weight vector ewt can be obtained
146 * through a call to the routine IDAGetErrWeights.
147 *
148 * user_data is a pointer to user data, the same as the user_data
149 * parameter passed to IDASetUserData.
150 *
151 * tmp is an N_Vector which can be used by the PrecSolve
152 * routine as temporary storage or work space.
153 *
154 * The IDASpilsPrecSolveFn should return
155 * 0 if successful,
156 * a positive int if a recoverable error occurred, or
157 * a negative int if a nonrecoverable error occurred.
158 * Following a recoverable error, the integrator will attempt to
159 * recover by updating the preconditioner and/or reducing the
160 * stepsize.
161 * -----------------------------------------------------------------
162 */
163
164typedef int (*IDASpilsPrecSolveFn)(realtype tt,
165 N_Vector yy, N_Vector yp, N_Vector rr,
166 N_Vector rvec, N_Vector zvec,
167 realtype c_j, realtype delta, void *user_data,
168 N_Vector tmp);

170/*
171 * -----------------------------------------------------------------
172 * Type : IDASpilsJacTimesVecFn
173 * -----------------------------------------------------------------
174 * The user-supplied function jtimes is to generate the product
175 * J*v for given v, where J is the Jacobian matrix
176 * J = dF/dy + c_j*dF/dy'
177 * or an approximation to it, and v is a given vector.
178 * It should return 0 if successful and a nonzero int otherwise.
179 *
180 * A function jtimes must have the prototype given below. Its
181 * parameters are as follows:
182 *
183 * tt is the current value of the independent variable.
184 *
185 * yy is the current value of the dependent variable vector,
186 * namely the predicted value of y(t).
187 *
188 * yp is the current value of the derivative vector y',
189 * namely the predicted value of y'(t).
190 *
191 * rr is the current value of the residual vector F(t,y,y').
192 *
193 * v is the N_Vector to be multiplied by J.
194 *
195 * Jv is the output N_Vector containing J*v.
196 *
197 * c_j is the scalar in the system Jacobian, proportional
198 * to 1/hh.
199 *
200 * user_data is a pointer to user data, the same as the
201 * pointer passed to IDASetUserData.
202 *
203 * tmp1, tmp2 are two N_Vectors which can be used by Jtimes for
204 * work space.
205 * -----------------------------------------------------------------
206 */
207
208typedef int (*IDASpilsJacTimesVecFn)(realtype tt,
209 N_Vector yy, N_Vector yp, N_Vector rr,
210 N_Vector v, N_Vector Jv,
211 realtype c_j, void *user_data,
212 N_Vector tmp1, N_Vector tmp2);

215/*
216 * -----------------------------------------------------------------
217 * Optional inputs to the IDASPILS linear solver
218 * -----------------------------------------------------------------
219 *
220 * IDASpilsSetPreconditioner specifies the PrecSetup and PrecSolve
221 * functions.
222 * Default is NULL for both arguments.
223 * IDASpilsSetJacTimesVecFn specifies the jtimes function.
224 * Default is to use an internal finite difference
225 * approximation routine.
226 * IDASpilsSetGSType specifies the type of Gram-Schmidt
227 * orthogonalization to be used. This must be one of
228 * the two enumeration constants MODIFIED_GS or
229 * CLASSICAL_GS defined in iterativ.h. These correspond
230 * to using modified Gram-Schmidt and classical
231 * Gram-Schmidt, respectively.
232 * Default value is MODIFIED_GS.
233 * Only for IDASPGMR.
234 * IDASpilsSetMaxRestarts specifies the maximum number of restarts
235 * to be used in the GMRES algorithm. maxrs must be a
236 * non-negative integer. Pass 0 to specify no restarts.
237 * Default is 5.
238 * Only for IDASPGMR.
239 * IDASpbcgSetMaxl specifies the maximum Krylov subspace size.
240 * Default is 5.
241 * Only for IDASPBCG and IDASPTFQMR.
242 * IDASpilsSetEpsLin specifies the factor in the linear iteration
243 * convergence test constant.
244 * Default is 0.05
245 * IDASpilsSetIncrementFactor specifies a factor in the increments
246 * to yy used in the difference quotient approximations
247 * to matrix-vector products Jv.
248 * Default is 1.0
249 *
250 * The return value of IDASpilsSet* is one of:
251 * IDASPILS_SUCCESS if successful
252 * IDASPILS_MEM_NULL if the ida memory was NULL
253 * IDASPILS_LMEM_NULL if the linear solver memory was NULL
254 * -----------------------------------------------------------------
255 */
