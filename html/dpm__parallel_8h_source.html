332/* lock for omp */
333#define MPT_OMP_LOCK(name, lockptr) \
334 { \
335 double _before = mpt_omp_lock_trace?MPT_wall_time():0.0; \
336 START_OMP_LOCK_PROFILE(_before); \
337 omp_set_lock(lockptr); \
338 if(mpt_omp_lock_trace){ \
339 mpt_omp_lock_time_start[MPT_LOCK_##name] = _before; \
340 mpt_omp_lock_wait_time[MPT_LOCK_##name] += \
341 MPT_wall_time()-mpt_omp_lock_time_start[MPT_LOCK_##name]; \
342 mpt_omp_lock_count[MPT_LOCK_##name]++;} \
343 }
344
345#define MPT_OMP_UNLOCK(name, lockptr) \
346 { \
347 if(mpt_omp_lock_trace){ \
348 mpt_omp_lock_time[MPT_LOCK_##name] += \
349 MPT_wall_time()-mpt_omp_lock_time_start[MPT_LOCK_##name];} \
350 omp_unset_lock(lockptr); \
351 STOP_OMP_LOCK_PROFILE; \
352 }
