306/* utility to loop over all (live and empty) sub-octrees in the forest */
307#define begin_forest_loop_all(forest, octree)\
308 {\
309 int __nsub_octrees = (forest)->num_sub_octrees;\
310 int __iter;\
311 intloop( __iter, __nsub_octrees)\
312 {\
313 octree = (forest)->sub_octrees[__iter];
314
315#define end_forest_loop_all(forest, octree)\
316 }}
317
318/* loop through all empty octrees */
319#define begin_forest_loop_empty(forest, octree)\
320 begin_forest_loop_all(forest, octree) \
321 if(!NULLP((octree)->octree_current))\
322 continue;
323
324#define end_forest_loop_empty(forest, octree)\
325 end_forest_loop_all(forest, octree)
326
327/* loop through all live octrees */
328#define begin_forest_loop(forest, octree)\
329 begin_forest_loop_all(forest, octree) \
330 if(NULLP((octree)->octree_current))\
331 continue;
332
333#define end_forest_loop(forest, octree)\
334 end_forest_loop_all(forest, octree)
335
336/* loop through leaf ptr octants of a pointer octree
337 * ltodo should be freed before 'jumping' out of the body of the loop
338 * (e.g. returning)*/
339#define begin_oct_loop(oct, ltodo, octree) \
340{ \
341 ltodo = L_NIL; \
342 oct = tg_ptr_octree_get_root_octant(octree); \
343 if (!NULLP(oct)) \
344 { \
345 L_PUSH(oct, ltodo); \
346 while (!NULLP(oct = (TgPtrOctant *)L_POP(ltodo))) \
347 { \
348 if (tg_ptr_octant_is_refined(oct)) \
349 { \
350 int _n##c; \
351 for (_n##c = 0; _n##c < TG_OCTANT_NUM_VERTS; _n##c++) \
352 L_PUSH(tg_ptr_octant_get_child(oct, _n##c), ltodo); \
353 } \
354 else \
355 {
356
357#define end_oct_loop(octree, ltodo, oct) \
358 } \
359 } \
360 } \
361 L_FREELIST(ltodo); \
362}
