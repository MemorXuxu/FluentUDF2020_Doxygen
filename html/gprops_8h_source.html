74/* real Opposite_Sides(Cell *, int, real *)
75 Returns positive value if the ith face is between the the ith node and
76 the vector, negative if they are on the same side, and zero if one is on
77 the face line.
78 To determine if two points are on opposite sides of a line or plane,
79 calculate the area of the triangles (or tetrahedrons) formed by the
80 line/plane and the two points. If the areas/volumes are have the same
81 sign then the points are on the same side.
82 We can eliminate the calculation of the area/volume of the cell since
83 the nodes are ordered to give a positive size.
84*/

259# define Face_Normal(f, nrm) \
260do { \
261 real NV_VEC(_E0), NV_VEC(_E1); \
262 real *_V0, *_V1; \
263 int _I; \
264 switch (FACE_TYPE(f)) \
265 { \
266 case TRI_FACE: \
267 NV_VV(_E0, =, FACE_NODE_COORD(f,1), -, FACE_NODE_COORD(f,0)); \
268 NV_VV(_E1, =, FACE_NODE_COORD(f,2), -, FACE_NODE_COORD(f,0)); \
269 NV_CROSS(nrm, =, _E0, _E1); \
270 break; \
271\
272 case QUAD_FACE: \
273 NV_VV(_E0, =, FACE_NODE_COORD(f,2), -, FACE_NODE_COORD(f,0)); \
274 NV_VV(_E1, =, FACE_NODE_COORD(f,3), -, FACE_NODE_COORD(f,1)); \
275 NV_CROSS(nrm, =, _E0, _E1); \
276 break; \
277\
278 default: \
279 NV_S(nrm, =, 0.0); \
280 face_node_loop(f,_I) \
281 { \
282 _V0 = FACE_NODE_COORD(f,_I); \
283 _V1 = FACE_NODE_COORD(f,NEXT_INDEX(FACE_NNODES(f),_I)); \
284 nrm[0] += ((_V1[1] + _V0[1]) * (_V1[2] - _V0[2])); \
285 nrm[1] += ((_V1[2] + _V0[2]) * (_V1[0] - _V0[0])); \
286 nrm[2] += ((_V1[0] + _V0[0]) * (_V1[1] - _V0[1])); \
287 } \
