163 void _ServeOneRequest(int timeout, TimeOutType toType)
164 {
165 for (;;)
166 {
167 RpcHandle handle = ansipc_Listen(0, timeout, toType);
168
169 CheckForError(handle);
170
171 if (handle == _rpcServer->Handle)
172 {
173 RpcHandle channel = OnNewConnection();
174 if (channel > 0)
175 {
176 if (_rpcServer->Mode == PlayBack)
177 RpcResetChannelRecordingAssociation(_rpcServer, channel);
178
179 break;
180 }
181 }
182 else if (ansipc_GetExtData(handle) == _rpcServer)
183 {
184 int eof = ansipc_CheckForEof(handle);
185 if (eof == ANSIPC_CON_DISCONNECT_CLEAN || eof == ANSIPC_CON_DISCONNECT_DIRTY) // a channel is unavailable: closed by peer gracefully or client exits without clean up
186 {
187 OnConnectionClosed(handle);
188 CheckForError((eof == ANSIPC_CON_DISCONNECT_CLEAN)? ANSIPC_ERROR_DISCONNECT_CLEAN : ANSIPC_ERROR_DISCONNECT_DIRTY, handle); //always throw
189 }
190 else
191 OnClientRequest(handle);
192
193 break;
194 }
195 }
196 }

268 static void CallCRequestHandler(RpcRequestHandler requestHandler, Request& request)
269 {
270 RpcRequest* rpcRequest = request.ToRpcRequest();

272
273 if (!(*requestHandler)(rpcRequest, rpcResponse))
274 rpcResponse->ErrorCode = kRpcErrorMethodNotFound;

282 void CallServantRequestHandler(IInterfaceServant* servant, Request& request)
283 {
284 Response response;
285
286 try
287 {
288 servant->HandleRequest(request, response);
289 }
290 catch (std::exception& ex)

292 response.ErrorCode = kRpcErrorGeneral;
293 response.AddArg<std::string>(typeid(ex).name());
294 response.AddArg<std::string>(ex.what());
295 }
296 catch (...)
297 {
298 response.ErrorCode = kRpcErrorGeneral;
299 }
300
301 response.Write(request.GetChannel());
302 }

304 RpcHandle OnNewConnection()
305 {
306 RpcHandle channel = _RpcAcceptServerConnection(_rpcServer);
307 CheckForError(channel);
308
309 if (channel > 0)
310 {
311 for (size_t i = 0; i < _connectionObservers.size(); i++)
312 if (!_connectionObservers[i]->OnConnectionOpened(channel))
313 {
314 _RpcCloseServerConnection(_rpcServer, channel);
315 break;
316 }
317 }
318
319 return channel;
320 }
321
322 void OnConnectionClosed(RpcHandle channel)
323 {
324 for (size_t i = 0; i < _connectionObservers.size(); i++)
325 _connectionObservers[i]->OnConnectionClosed(channel);
326
327 _RpcCloseServerConnection(_rpcServer, channel);
328 }

330 virtual void OnClientRequest(RpcHandle channel)
331 {
332 Request request;
333
334 request.Read(channel);
335
336 RpcRequestHandler requestHandler = GetCRequestHandler(request.GetInterface());
337 if (requestHandler != NULL)
338 {
339 CallCRequestHandler(requestHandler, request);
340 return;
341 }
342
343 IInterfaceServant* servant = GetServant(request.GetInterface());
344 if (servant != NULL)
345 {
346 CallServantRequestHandler(servant, request);
347 return;
348 }
349
350 Response response(kRpcErrorInterfaceNotFound);
351 response.Write(channel);
352 }
