61# define PRF_CSR_N_INT_N 1000
62# define PRF_CSEND_N_INT(i, n, d, t) \
63 { \
64 int todo = *(n); \
65 int mybuff[PRF_CSR_N_INT_N + 1]; \
66 *mybuff = todo; \
67 if (todo > PRF_CSR_N_INT_N) \
68 todo = PRF_CSR_N_INT_N; \
69 if (0 < todo) \
70 memcpy(mybuff + 1, (d), todo * sizeof(int)); \
71 if (todo < PRF_CSR_N_INT_N) \
72 memset((char *)((int *)mybuff + 1 + todo), \
73 0, (PRF_CSR_N_INT_N - todo) * sizeof (int)); \
74 PRF_CSEND_INT((i), mybuff, PRF_CSR_N_INT_N + 1, (t)); \
75 if (*(n) > PRF_CSR_N_INT_N) \
76 PRF_CSEND_INT((i), (d) + PRF_CSR_N_INT_N, *(n) - PRF_CSR_N_INT_N, (t)); \
77 }
78
79#define PRF_CRECV_N_INT(i, n, d, t) \
80 { \
81 int todo; \
82 int mybuff[PRF_CSR_N_INT_N + 1]; \
83 PRF_CRECV_INT((i), mybuff, PRF_CSR_N_INT_N + 1, (t)); \
84 *(n) = *mybuff; \
85 (d) = CX_Realloc((d), sizeof(int) * *mybuff); \
86 todo = *mybuff; \
87 if (todo > PRF_CSR_N_INT_N) \
88 todo = PRF_CSR_N_INT_N; \
89 if (0 < todo) \
90 memcpy((d), mybuff + 1, todo * sizeof(int)); \
91 if (*mybuff > PRF_CSR_N_INT_N) \
92 PRF_CRECV_INT((i), (d) + PRF_CSR_N_INT_N, *mybuff - PRF_CSR_N_INT_N, (t)); \
93 }
