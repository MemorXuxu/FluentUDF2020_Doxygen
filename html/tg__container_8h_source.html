126/*push back must be a macro to avoid the mem_block to be casted*/
127#define _TG_VECTOR_PUSH_BACK(mem_block, val) \
128 { \
129 (mem_block) = _TG_Vector_Check_Resize((void*)(mem_block)); \
130 (mem_block)[_TG_VECTOR_SIZE((void*)(mem_block))] = (val); \
131 _TG_VECTOR_SIZE((void*)(mem_block))++; \
132 }
133
134/* pop */
135#define _TG_VECTOR_POP(mem_block, obj)\
136 ((_TG_VECTOR_SIZE((void*)mem_block) == 0)? \
137 ((obj = NULL), FALSE) : \
138 ((obj = mem_block[--(_TG_VECTOR_SIZE((void*)mem_block))]), TRUE))

140/*return a pointer to the first free slot.

144 { \
145 mem_block = _TG_Vector_Check_Resize((void*)mem_block); \
146 out_ptr = &mem_block[_TG_VECTOR_SIZE(mem_block)] ; \
147 _TG_VECTOR_SIZE(mem_block)++; \
148 }
