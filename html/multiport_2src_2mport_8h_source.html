875/* Wrappers for serializing a piece of code for parallel runs.
876 * The enclosed piece of code will be executed first by the host,
877 * then node-0, node-1, node-2, and so on.
878 *
879 * BEGIN_SERIALIZE_CODE();
880 *
881 * statement;
882 * ...
883 *
884 * END_SERIALIZE_CODE();
885 */
886#define BEGIN_SERIALIZE_CODE() \
887 { int _baton=1; \
888 int _myid = MPT_Myid(); \
889 int _from = (MPT_I_AM_NODE_ZERO_P ? MPID_HOST : _myid-1); \
890 int _to = (MPT_I_AM_NODE_HOST_P ? mpt_node_zero : _myid+1); \
891 if (!MPT_I_AM_NODE_HOST_P) \
892 MPT_CRECV_INT(_from, &_baton, 1, _from);
893
894#define END_SERIALIZE_CODE() \
895 if (!MPT_I_AM_NODE_LAST_P) \
896 MPT_CSEND_INT(_to, &_baton, 1, _myid); \
897 }
898
899#define BEGIN_SERIALIZE_NODES() \
900 if (MPT_I_AM_COMPUTE_NODE_P) \
901 { int _baton=1; \
902 int _myid = MPT_Myid(); \
903 if (!MPT_I_AM_NODE_ZERO_P) \
904 MPT_CRECV_INT(_myid-1, &_baton, 1, _myid-1);
905
906#define END_SERIALIZE_NODES() \
907 if (!MPT_I_AM_NODE_LAST_P) \
908 MPT_CSEND_INT(_myid+1, &_baton, 1, _myid); \
909 }
