99/*
100 Note 1:
101 ------
102 Please note that I have modified this macro, because previous version of this has following issues:
103 i) CR 73778
104 ii) For the cases where we have solid + fluid region both & if "thread_loop_c" starts with the solid
105 thread as first_solve_thread then, this loop will only populate the SV_T gradient & will not
106 populate the other scalar gradient which are solved in coupled manner, like SV_P ...etc.
107
108 Note 2:
109 ------
110 Please make sure follwing while using this macro:
111 This macro take care for the cases where we have
112 i) Only Fluid Zone
113 ii) Fluid & Solid Zone both
114
115 But it don't take care of the cases where we have only Solid Zone, so one need to put an "else" statement
116 to handle them. For example:
117
118 begin_domain_eqn_loop(eqn,sv,domain)
119 {
120 //this is a very important statement we have to use this
121 //always to prevent the parallel issue found in CR 73778
122 sv = PRF_GIHIGH(sv);
123 Repain_Gradient(domain,sv,SV_EQN_RG(sv));
124 }
125 else
126 Repain_Gradient(domain,SV_T,SV_EQN_RG(SV_T));
127 end_domain_eqn_loop(eqn,sv,domain);
128*/
129#define begin_domain_eqn_loop(eqn,sv,d) \
130{ \
131 int _domain_nr_equations=0; \
132 Thread* _first_solve_thread=NULL, *_thread; \
133 cxboolean has_fluid = FALSE; \
134 thread_loop_c(_thread,d) \
135 { \
136 _domain_nr_equations=THREAD_N_EQNS(_thread); \
137 _domain_nr_equations=PRF_GIHIGH1(_domain_nr_equations); \
138 if((_domain_nr_equations!=0)&& FLUID_THREAD_P(_thread)){ \
139 _first_solve_thread=_thread; \
140 has_fluid = TRUE; \
141 break; \
142 } \
143 } \
144 if(PRF_ANYONE(has_fluid,FALSE)) \
145 bounded_eqn_loop(0,_domain_nr_equations,eqn,sv,_first_solve_thread)
146
147#define end_domain_eqn_loop(eqn,sv,d) }
