10* NAME: solpos.h
11*
12* Contains:
13* S_date (computes the date from the day number)
14* INPUTS: S_year, S_daynum
15* OUTPUTS: S_year, S_month, S_day
16*
17* S_solpos (computes the solar position and intensity
18* from time and place)
19* INPUTS: S_month, S_day, S_latitude, S_longitude, S_timezone
20* OPTIONAL:
21* S_year DEFAULT 2001 (non-leap)
22* S_hour DEFAULT 12 (noon)
23* S_minute DEFAULT 0 ( " )
24* S_second DEFAULT 0 ( " )
25* S_press DEFAULT 1013.0 (standard pressure)
26* S_temp DEFAULT 10.0 (standard temperature)
27* S_tilt DEFAULT 0.0 (horizontal panel)
28* S_aspect DEFAULT 180.0 (South-facing panel)
29*
30* OUTPUTS: S_daynum, S_amass, S_ampress, S_azim, S_cosinc,
31* S_elevref, S_etr, S_etrn, S_etrtilt, S_prime,
32* S_sbcf, S_sunrise, S_sunset, S_unprime, S_zenref
33*
34* Usage:
35* In calling program, along with other 'includes', insert:
36*
37* #include "solpos.h"

39* Martin Rymes

130/*============================================================================
131* Integer function S_date, adapted from the NREL VAX solar libraries
132*
133* This function calculates the month and day of month, given the
134* day number (number of day in year, sometimes confusingly called
135* the "Julian Day", but I know that you will never call it that).
136*
137* Requires:
138* Year and day number:
139* S_year NOTE: 2-digit year "xx" is assumed "19xx"
140* DEFAULT S_year = 2001
141* S_daynum RANGE: -1 to a large positive integer
142*
143* Returns:
144* Year, month, day:
145* S_year
146* S_month
147* S_day
148*----------------------------------------------------------------------------*/
149extern int S_date (void);
150
151/*============================================================================
152* Integer function S_solpos, adapted from the NREL VAX solar libraries
153*
154* This function calculates the apparent solar position and intensity
155* (theoretical maximum solar energy) based on the date, time, and
156* location on Earth.
157*
158* Requires:
159* Date and time:
160* S_year DEFAULT 2001
161* S_month
162* S_day
163* S_hour DEFAULT 12
164* S_minute DEFAULT 0
165* S_second DEFAULT 0
166* Location:
167* S_latitude
168* S_longitude
169* Location/time adjuster:
170* S_timezone
171* Atmospheric pressure and temperature:
172* S_press DEFAULT 1013.0 mb
173* S_temp DEFAULT 10.0 degrees C
174* Tilt of flat surface that receives solar energy:
175* S_aspect DEFAULT 180 (South)
176* S_tilt DEFAULT 0 (Horizontal)
177*
178* Returns:
179* everything defined at the top of this listing.
180*----------------------------------------------------------------------------*/
181extern int S_solpos (void);
