65// WIDTHBYTES performs DWORD-aligning of DIB scanlines. The "bits"
66// parameter is the bit count for the scanline (biWidth * biBitCount),
67// and this macro returns the number of DWORD-aligned bytes needed
68// to hold those bits.
69
70#define WIDTHBYTES(bits) (((bits) + 31) / 32 * 4)
