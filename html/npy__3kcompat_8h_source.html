164/*
165 * Get a FILE* handle to the file represented by the Python object
166 */
167static NPY_INLINE FILE*
168npy_PyFile_Dup2(PyObject *file, char *mode, npy_off_t *orig_pos)
169{
170 int fd, fd2, unbuf;
171 PyObject *ret, *os, *io, *io_raw;
172 npy_off_t pos;
173 FILE *handle;
174
175 /* For Python 2 PyFileObject, use PyFile_AsFile */
176#if !defined(NPY_PY3K)
177 if (PyFile_Check(file)) {
178 return PyFile_AsFile(file);
179 }
180#endif
181
182 /* Flush first to ensure things end up in the file in the correct order */
183 ret = PyObject_CallMethod(file, "flush", "");
184 if (ret == NULL) {
185 return NULL;
186 }
187 Py_DECREF(ret);
188 fd = PyObject_AsFileDescriptor(file);
189 if (fd == -1) {
190 return NULL;
191 }
192
193 /*
194 * The handle needs to be dup'd because we have to call fclose
195 * at the end
196 */
197 os = PyImport_ImportModule("os");
198 if (os == NULL) {
199 return NULL;
200 }
201 ret = PyObject_CallMethod(os, "dup", "i", fd);
202 Py_DECREF(os);
203 if (ret == NULL) {
204 return NULL;
205 }
206 fd2 = PyNumber_AsSsize_t(ret, NULL);
207 Py_DECREF(ret);
208
209 /* Convert to FILE* handle */
210#ifdef _WIN32
211 handle = _fdopen(fd2, mode);
212#else
213 handle = fdopen(fd2, mode);
214#endif
215 if (handle == NULL) {
216 PyErr_SetString(PyExc_IOError,
217 "Getting a FILE* from a Python file object failed");
218 }
219
220 /* Record the original raw file handle position */
221 *orig_pos = npy_ftell(handle);
222 if (*orig_pos == -1) {
223 /* The io module is needed to determine if buffering is used */
224 io = PyImport_ImportModule("io");
225 if (io == NULL) {
226 fclose(handle);
227 return NULL;
228 }
229 /* File object instances of RawIOBase are unbuffered */
230 io_raw = PyObject_GetAttrString(io, "RawIOBase");
231 Py_DECREF(io);
232 if (io_raw == NULL) {
233 fclose(handle);
234 return NULL;
235 }
236 unbuf = PyObject_IsInstance(file, io_raw);
237 Py_DECREF(io_raw);
238 if (unbuf == 1) {
239 /* Succeed if the IO is unbuffered */
240 return handle;
241 }
242 else {
243 PyErr_SetString(PyExc_IOError, "obtaining file position failed");
244 fclose(handle);
245 return NULL;
246 }
247 }
248
249 /* Seek raw handle to the Python-side position */
250 ret = PyObject_CallMethod(file, "tell", "");
251 if (ret == NULL) {
252 fclose(handle);
253 return NULL;
254 }
255 pos = PyLong_AsLongLong(ret);
256 Py_DECREF(ret);
257 if (PyErr_Occurred()) {
258 fclose(handle);
259 return NULL;
260 }
261 if (npy_fseek(handle, pos, SEEK_SET) == -1) {
262 PyErr_SetString(PyExc_IOError, "seeking file failed");
263 fclose(handle);
264 return NULL;
265 }
266 return handle;
267}

269/*
270 * Close the dup-ed file handle, and seek the Python one to the current position
271 */
272static NPY_INLINE int
273npy_PyFile_DupClose2(PyObject *file, FILE* handle, npy_off_t orig_pos)
274{
275 int fd, unbuf;
276 PyObject *ret, *io, *io_raw;
277 npy_off_t position;
278
279 /* For Python 2 PyFileObject, do nothing */
280#if !defined(NPY_PY3K)
281 if (PyFile_Check(file)) {
282 return 0;
283 }
284#endif
285
286 position = npy_ftell(handle);
287
288 /* Close the FILE* handle */
289 fclose(handle);
290
291 /*
292 * Restore original file handle position, in order to not confuse
293 * Python-side data structures
294 */
295 fd = PyObject_AsFileDescriptor(file);
296 if (fd == -1) {
297 return -1;
298 }
299
300 if (npy_lseek(fd, orig_pos, SEEK_SET) == -1) {
301
302 /* The io module is needed to determine if buffering is used */
303 io = PyImport_ImportModule("io");
304 if (io == NULL) {
305 return -1;
306 }
307 /* File object instances of RawIOBase are unbuffered */
308 io_raw = PyObject_GetAttrString(io, "RawIOBase");
309 Py_DECREF(io);
310 if (io_raw == NULL) {
311 return -1;
312 }
313 unbuf = PyObject_IsInstance(file, io_raw);
314 Py_DECREF(io_raw);
315 if (unbuf == 1) {
316 /* Succeed if the IO is unbuffered */
317 return 0;
318 }
319 else {
320 PyErr_SetString(PyExc_IOError, "seeking file failed");
321 return -1;
322 }
323 }
324
325 if (position == -1) {
326 PyErr_SetString(PyExc_IOError, "obtaining file position failed");
327 return -1;
328 }
329
330 /* Seek Python-side handle to the FILE* handle position */
331 ret = PyObject_CallMethod(file, "seek", NPY_OFF_T_PYFMT "i", position, 0);
332 if (ret == NULL) {
333 return -1;
334 }
335 Py_DECREF(ret);
336 return 0;
337}

340npy_PyFile_Check(PyObject *file)
341{
342 int fd;
343 /* For Python 2, check if it is a PyFileObject */
344#if !defined(NPY_PY3K)
345 if (PyFile_Check(file)) {
346 return 1;
347 }
348#endif
349 fd = PyObject_AsFileDescriptor(file);
350 if (fd == -1) {
351 PyErr_Clear();
352 return 0;
353 }
354 return 1;

385static NPY_INLINE int
386PyObject_Cmp(PyObject *i1, PyObject *i2, int *cmp)
387{
388 int v;
389 v = PyObject_RichCompareBool(i1, i2, Py_LT);
390 if (v == 1) {
391 *cmp = -1;
392 return 1;
393 }
394 else if (v == -1) {
395 return -1;
396 }
397
398 v = PyObject_RichCompareBool(i1, i2, Py_GT);
399 if (v == 1) {
400 *cmp = 1;
401 return 1;
402 }
403 else if (v == -1) {
404 return -1;
405 }
406
407 v = PyObject_RichCompareBool(i1, i2, Py_EQ);
408 if (v == 1) {
409 *cmp = 0;
410 return 1;
411 }
412 else {
413 *cmp = 0;
414 return -1;
415 }
416}
