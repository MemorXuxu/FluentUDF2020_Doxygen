50 area[0] = 0.5*((v1[1]-v0[1])*(v2[2]-v0[2]) - (v2[1]-v0[1])*(v1[2]-v0[2]));\
51 area[1] = 0.5*((v1[2]-v0[2])*(v2[0]-v0[0]) - (v2[2]-v0[2])*(v1[0]-v0[0]));\
