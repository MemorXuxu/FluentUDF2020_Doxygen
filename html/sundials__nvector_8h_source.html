13 * This is the header file for a generic NVECTOR package.
14 * It defines the N_Vector structure (_generic_N_Vector) which
15 * contains the following fields:
16 * - an implementation-dependent 'content' field which contains
17 * the description and actual data of the vector
18 * - an 'ops' filed which contains a structure listing operations
19 * acting on such vectors
20 *
21 * Part I of this file contains type declarations for the
22 * _generic_N_Vector and _generic_N_Vector_Ops structures, as well
23 * as references to pointers to such structures (N_Vector).
24 *
25 * Part II of this file contains the prototypes for the vector
26 * functions which operate on N_Vector.
27 *
28 * At a minimum, a particular implementation of an NVECTOR must
29 * do the following:
30 * - specify the 'content' field of N_Vector,
31 * - implement the operations on those N_Vectors,
32 * - provide a constructor routine for new vectors
33 *
34 * Additionally, an NVECTOR implementation may provide the following:
35 * - macros to access the underlying N_Vector data
36 * - a constructor for an array of N_Vectors
37 * - a constructor for an empty N_Vector (i.e., a new N_Vector with
38 * a NULL data pointer).
39 * - a routine to print the content of an N_Vector
40 * -----------------------------------------------------------------
41 */

115/*
116 * -----------------------------------------------------------------
117 * N_VClone
118 * Creates a new vector of the same type as an existing vector.
119 * It does not copy the vector, but rather allocates storage for
120 * the new vector.
121 *
122 * N_VCloneEmpty
123 * Creates a new vector of the same type as an existing vector,
124 * but does not allocate storage.
125 *
126 * N_VDestroy
127 * Destroys a vector created with N_VClone.
128 *
129 * N_VSpace
130 * Returns space requirements for one N_Vector (type 'realtype' in
131 * lrw and type 'long int' in liw).
132 *
133 * N_VGetArrayPointer
134 * Returns a pointer to the data component of the given N_Vector.
135 * NOTE: This function assumes that the internal data is stored
136 * as a contiguous 'realtype' array. This routine is only used in
137 * the solver-specific interfaces to the dense and banded linear
138 * solvers, as well as the interfaces to the banded preconditioners
139 * distributed with SUNDIALS.
140 *
141 * N_VSetArrayPointer
142 * Overwrites the data field in the given N_Vector with a user-supplied
143 * array of type 'realtype'.
144 * NOTE: This function assumes that the internal data is stored
145 * as a contiguous 'realtype' array. This routine is only used in
146 * the interfaces to the dense linear solver.
147 *
148 * N_VLinearSum
149 * Performs the operation z = a*x + b*y
150 *
151 * N_VConst
152 * Performs the operation z[i] = c for i = 0, 1, ..., N-1
153 *
154 * N_VProd
155 * Performs the operation z[i] = x[i]*y[i] for i = 0, 1, ..., N-1
156 *
157 * N_VDiv
158 * Performs the operation z[i] = x[i]/y[i] for i = 0, 1, ..., N-1
159 *
160 * N_VScale
161 * Performs the operation z = c*x
162 *
163 * N_VAbs
164 * Performs the operation z[i] = |x[i]| for i = 0, 1, ..., N-1
165 *
166 * N_VInv
167 * Performs the operation z[i] = 1/x[i] for i = 0, 1, ..., N-1
168 * This routine does not check for division by 0. It should be
169 * called only with an N_Vector x which is guaranteed to have
170 * all non-zero components.
171 *
172 * N_VAddConst
173 * Performs the operation z[i] = x[i] + b for i = 0, 1, ..., N-1
174 *
175 * N_VDotProd
176 * Returns the dot product of two vectors:
177 * sum (i = 0 to N-1) {x[i]*y[i]}
178 *
179 * N_VMaxNorm
180 * Returns the maximum norm of x:
181 * max (i = 0 to N-1) ABS(x[i])
182 *
183 * N_VWrmsNorm
184 * Returns the weighted root mean square norm of x with weight
185 * vector w:
186 * sqrt [(sum (i = 0 to N-1) {(x[i]*w[i])^2})/N]
187 *
188 * N_VWrmsNormMask
189 * Returns the weighted root mean square norm of x with weight
190 * vector w, masked by the elements of id:
191 * sqrt [(sum (i = 0 to N-1) {(x[i]*w[i]*msk[i])^2})/N]
192 * where msk[i] = 1.0 if id[i] > 0 and
193 * msk[i] = 0.0 if id[i] < 0
194 *
195 * N_VMin
196 * Returns the smallest element of x:
197 * min (i = 0 to N-1) x[i]
198 *
199 * N_VWL2Norm
200 * Returns the weighted Euclidean L2 norm of x with weight
201 * vector w:
202 * sqrt [(sum (i = 0 to N-1) {(x[i]*w[i])^2})]
203 *
204 * N_VL1Norm
205 * Returns the L1 norm of x:
206 * sum (i = 0 to N-1) {ABS(x[i])}
207 *
208 * N_VCompare
209 * Performs the operation
210 * z[i] = 1.0 if ABS(x[i]) >= c i = 0, 1, ..., N-1
211 * 0.0 otherwise
212 *
213 * N_VInvTest
214 * Performs the operation z[i] = 1/x[i] with a test for
215 * x[i] == 0.0 before inverting x[i].
216 * This routine returns TRUE if all components of x are non-zero
217 * (successful inversion) and returns FALSE otherwise.
218 *
219 * N_VConstrMask
220 * Performs the operation :
221 * m[i] = 1.0 if constraint test fails for x[i]
222 * m[i] = 0.0 if constraint test passes for x[i]
223 * where the constraint tests are as follows:
224 * If c[i] = +2.0, then x[i] must be > 0.0.
225 * If c[i] = +1.0, then x[i] must be >= 0.0.
226 * If c[i] = -1.0, then x[i] must be <= 0.0.
227 * If c[i] = -2.0, then x[i] must be < 0.0.
228 * This routine returns a boolean FALSE if any element failed
229 * the constraint test, TRUE if all passed. It also sets a
230 * mask vector m, with elements equal to 1.0 where the
231 * corresponding constraint test failed, and equal to 0.0
232 * where the constraint test passed.
233 * This routine is specialized in that it is used only for
234 * constraint checking.
235 *
236 * N_VMinQuotient
237 * Performs the operation :
238 * minq = min ( num[i]/denom[i]) over all i such that
239 * denom[i] != 0.
240 * This routine returns the minimum of the quotients obtained
241 * by term-wise dividing num[i] by denom[i]. A zero element
242 * in denom will be skipped. If no such quotients are found,
243 * then the large value BIG_REAL is returned.
244 *
245 * -----------------------------------------------------------------
246 *
247 * The following table lists the vector functions used by
248 * different modules in SUNDIALS. The symbols in the table
249 * have the following meaning:
250 * S - called by the solver;
251 * D - called by the dense linear solver module
252 * B - called by the band linear solver module
253 * Di - called by the diagonal linear solver module
254 * I - called by the iterative linear solver module
255 * BP - called by the band preconditioner module
256 * BBDP - called by the band-block diagonal preconditioner module
257 * F - called by the Fortran-to-C interface
258 *
259 * ------------------------------------------------
260 * MODULES
261 * NVECTOR ------------------------------------------------
262 * FUNCTIONS CVODE/CVODES IDA KINSOL
263 * -----------------------------------------------------------------
264 * N_VClone S Di I S I BBDP S I BBDP
265 * -----------------------------------------------------------------
266 * N_VCloneEmpty F F F
267 * -----------------------------------------------------------------
268 * N_VDestroy S Di I S I BBDP S I BBDP
269 * -----------------------------------------------------------------
270 * N_VSpace S S S
271 * -----------------------------------------------------------------
272 * N_VGetArrayPointer D B BP BBDP F D B BBDP BBDP F
273 * -----------------------------------------------------------------
274 * N_VSetArrayPointer D F D F
275 * -----------------------------------------------------------------
276 * N_VLinearSum S D Di I S D I S I
277 * -----------------------------------------------------------------
278 * N_VConst S I S I I
279 * -----------------------------------------------------------------
280 * N_VProd S Di I S I S I
281 * -----------------------------------------------------------------
282 * N_VDiv S Di I S I S I
283 * -----------------------------------------------------------------
284 * N_VScale S D B Di I BP BBDP S D B I BBDP S I BBDP
285 * -----------------------------------------------------------------
286 * N_VAbs S S S
287 * -----------------------------------------------------------------
288 * N_VInv S Di S S
289 * -----------------------------------------------------------------
290 * N_VAddConst S Di S
291 * -----------------------------------------------------------------
292 * N_VDotProd I I I
293 * -----------------------------------------------------------------
294 * N_VMaxNorm S S S
295 * -----------------------------------------------------------------
296 * N_VWrmsNorm S D B I BP BBDP S
297 * -----------------------------------------------------------------
298 * N_VWrmsNormMask S
299 * -----------------------------------------------------------------
300 * N_VMin S S S
301 * -----------------------------------------------------------------
302 * N_VWL2Norm S I
303 * -----------------------------------------------------------------
304 * N_VL1Norm I
305 * -----------------------------------------------------------------
306 * N_VCompare Di S
307 * -----------------------------------------------------------------
308 * N_VInvTest Di
309 * -----------------------------------------------------------------
310 * N_VConstrMask S S
311 * -----------------------------------------------------------------
312 * N_VMinQuotient S S
313 * -----------------------------------------------------------------
314 */
