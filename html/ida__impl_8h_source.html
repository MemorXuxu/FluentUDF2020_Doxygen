254/*
255 * -----------------------------------------------------------------
256 * int (*ida_linit)(IDAMem IDA_mem);
257 * -----------------------------------------------------------------
258 * The purpose of ida_linit is to allocate memory for the
259 * solver-specific fields in the structure *(idamem->ida_lmem) and
260 * perform any needed initializations of solver-specific memory,
261 * such as counters/statistics. An (*ida_linit) should return
262 * 0 if it has successfully initialized the IDA linear solver and
263 * a non-zero value otherwise. If an error does occur, an appropriate
264 * message should be sent to the error handler function.
265 * ----------------------------------------------------------------
266 */
267
268/*
269 * -----------------------------------------------------------------
270 * int (*ida_lsetup)(IDAMem IDA_mem, N_Vector yyp, N_Vector ypp,
271 * N_Vector resp,
272 * N_Vector tempv1, N_Vector tempv2, N_Vector tempv3);
273 * -----------------------------------------------------------------
274 * The job of ida_lsetup is to prepare the linear solver for
275 * subsequent calls to ida_lsolve. Its parameters are as follows:
276 *
277 * idamem - problem memory pointer of type IDAMem. See the big
278 * typedef earlier in this file.
279 *
280 *
281 * yyp - the predicted y vector for the current IDA internal
282 * step.
283 *
284 * ypp - the predicted y' vector for the current IDA internal
285 * step.
286 *
287 * resp - F(tn, yyp, ypp).
288 *
289 * tempv1, tempv2, tempv3 - temporary N_Vectors provided for use
290 * by ida_lsetup.
291 *
292 * The ida_lsetup routine should return 0 if successful,
293 * a positive value for a recoverable error, and a negative value
294 * for an unrecoverable error.
295 * -----------------------------------------------------------------
296 */
297
298/*
299 * -----------------------------------------------------------------
300 * int (*ida_lsolve)(IDAMem IDA_mem, N_Vector b, N_Vector weight,
301 * N_Vector ycur, N_Vector ypcur, N_Vector rescur);
302 * -----------------------------------------------------------------
303 * ida_lsolve must solve the linear equation P x = b, where
304 * P is some approximation to the system Jacobian
305 * J = (dF/dy) + cj (dF/dy')
306 * evaluated at (tn,ycur,ypcur) and the RHS vector b is input.
307 * The N-vector ycur contains the solver's current approximation
308 * to y(tn), ypcur contains that for y'(tn), and the vector rescur
309 * contains the N-vector residual F(tn,ycur,ypcur).
310 * The solution is to be returned in the vector b.
311 *
312 * The ida_lsolve routine should return 0 if successful,
313 * a positive value for a recoverable error, and a negative value
314 * for an unrecoverable error.
315 * -----------------------------------------------------------------
316 */
317
318/*
319 * -----------------------------------------------------------------
320 * int (*ida_lperf)(IDAMem IDA_mem, int perftask);
321 * -----------------------------------------------------------------
322 * ida_lperf is called two places in IDA where linear solver
323 * performance data is required by IDA. For perftask = 0, an
324 * initialization of performance variables is performed, while for
325 * perftask = 1, the performance is evaluated.
326 * -----------------------------------------------------------------
327 */
328

330 * -----------------------------------------------------------------
331 * int (*ida_lfree)(IDAMem IDA_mem);
332 * -----------------------------------------------------------------
333 * ida_lfree should free up any memory allocated by the linear
334 * solver. This routine is called once a problem has been
335 * completed and the linear solver is no longer needed.
336 * -----------------------------------------------------------------
337 */
