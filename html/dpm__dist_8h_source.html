228/* 2012-01-19 ubecker During property calculation, we may access
229 * DPM_DIST_VAR storage before any distributions have been defined.
230 * Then the new d->svar points to SV_P and may or may not give a
231 * segmentation violation. But v->index is also garbage and so will
232 * be the return value. Need to guard against this: Only use
233 * distribution, if it is_defined.
234 *
235 * Snatch: DPM_DIST_VAR was also understood as a storage access and
236 * used as such in many places.
237 *
238 * Introduce new macro DPM_DIST_VAR_IF_DEFINED for protected storage
239 * access for places that may be accessed before the distribution
240 * setup occured, like in functions called directly from routines
241 * "elsewhere" in the code.
242 *
243 * "te" means thread element; i.e., node, face, or cell.
244 */
245#define DPM_DIST_VAR(te,t,d,v,i_class,i_dim) (T_STORAGE_R_XV(t,(d)->svar,t->dpm_dist[(v)->index] + i_class + i_dim * (d)->classification_data.n_class)[te])
246#define DPM_DIST_VAR_IF_DEFINED(et,t,d,v,i_class,i_dim) ((d)->is_defined ? DPM_DIST_VAR(et,t,d,v,i_class,i_dim) : 0)
