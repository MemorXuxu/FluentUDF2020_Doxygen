87/*
88 * -----------------------------------------------------------------
89 * Function : SpbcgSolve
90 * -----------------------------------------------------------------
91 * SpbcgSolve solves the linear system Ax = b by means of a scaled
92 * preconditioned Bi-CGSTAB (SPBCG) iterative method.
93 *
94 * mem pointer to an internal memory block allocated during a
95 * prior call to SpbcgMalloc
96 *
97 * A_data pointer to a data structure containing information
98 * about the coefficient matrix A (passed to user-supplied
99 * function referenced by atimes (function pointer))
100 *
101 * x vector (type N_Vector) containing initial guess x_0 upon
102 * entry, but which upon return contains an approximate solution
103 * of the linear system Ax = b (solution only valid if return
104 * value is either SPBCG_SUCCESS or SPBCG_RES_REDUCED)
105 *
106 * b vector (type N_Vector) set to the right-hand side vector b
107 * of the linear system (undisturbed by function)
108 *
109 * pretype variable (type int) indicating the type of
110 * preconditioning to be used (see sundials_iterative.h)
111 *
112 * delta tolerance on the L2 norm of the scaled, preconditioned
113 * residual (if return value == SPBCG_SUCCESS, then
114 * ||sb*P1_inv*(b-Ax)||_L2 <= delta)
115 *
116 * P_data pointer to a data structure containing preconditioner
117 * information (passed to user-supplied function referenced
118 * by psolve (function pointer))
119 *
120 * sx vector (type N_Vector) containing positive scaling factors
121 * for x (pass sx == NULL if scaling NOT required)
122 *
123 * sb vector (type N_Vector) containing positive scaling factors
124 * for b (pass sb == NULL if scaling NOT required)
125 *
126 * atimes user-supplied routine responsible for computing the
127 * matrix-vector product Ax (see sundials_iterative.h)
128 *
129 * psolve user-supplied routine responsible for solving the
130 * preconditioned linear system Pz = r (ignored if
131 * pretype == PREC_NONE) (see sundials_iterative.h)
132 *
133 * res_norm pointer (type realtype*) to the L2 norm of the
134 * scaled, preconditioned residual (if return value
135 * is either SPBCG_SUCCESS or SPBCG_RES_REDUCED, then
136 * *res_norm = ||sb*P1_inv*(b-Ax)||_L2, where x is
137 * the computed approximate solution, sb is the diagonal
138 * scaling matrix for the right-hand side b, and P1_inv
139 * is the inverse of the left-preconditioner matrix)
140 *
141 * nli pointer (type int*) to the total number of linear
142 * iterations performed
143 *
144 * nps pointer (type int*) to the total number of calls made
145 * to the psolve routine
146 * -----------------------------------------------------------------
147 */
148

155
156#define SPBCG_SUCCESS 0 /* SPBCG algorithm converged */
157#define SPBCG_RES_REDUCED 1 /* SPBCG did NOT converge, but the

159#define SPBCG_CONV_FAIL 2 /* SPBCG algorithm failed to converge */
160#define SPBCG_PSOLVE_FAIL_REC 3 /* psolve failed recoverably */
161#define SPBCG_ATIMES_FAIL_REC 4 /* atimes failed recoverably */
162#define SPBCG_PSET_FAIL_REC 5 /* pset faild recoverably */
163
164#define SPBCG_MEM_NULL -1 /* mem argument is NULL */
165#define SPBCG_ATIMES_FAIL_UNREC -2 /* atimes returned failure flag */
166#define SPBCG_PSOLVE_FAIL_UNREC -3 /* psolve failed unrecoverably */
