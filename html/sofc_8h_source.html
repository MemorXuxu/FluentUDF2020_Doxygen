254static int Find_Thread(Thread *t)
255{
256 Domain *domain = Get_Domain(1);
257 Thread *test;
258 int id = -1, index, i;
259
260 static Thread *last_thread;
261 static int last_id, count = -1;
262
263
264 /* First, check to see if the last lookup is the same */
265 if (t == last_thread) return last_id;
266
267 /* Otherwise, grind through the whole list. */
268
269 if (count == -1)
270 {
271 count = 0;
272 for (i = 0; i < 100000; i++)
273 {
274 if (Lookup_Thread(domain, i) != NULL)
275 count = i;
276 }
277 CX_Message("Total Thread Count: %d\n", count);
278 }
279
280
281 for (index = 0; index <= count; index++)
282 {
283 test = Lookup_Thread(domain, index);
284 if (test == t)
285 {
286 id = index;
287 last_thread = test;
288 last_id = index;
289 return id;
290 }
291 }
292
293 return id;
294}
